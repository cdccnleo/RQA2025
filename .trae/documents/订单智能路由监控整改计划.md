1. **路由监控器**
   ```python
   class OrderRoutingMonitor:
       """订单路由监控器"""
       
       ALERT_RULES = {
           'routing_failure': {
               'threshold': 0.05,  # 失败率超过5%
               'severity': 'high',
               'description': '路由失败率过高'
           },
           'high_latency': {
               'threshold': 100,  # 延迟超过100ms
               'severity': 'medium',
               'description': '路由延迟过高'
           },
           'low_success_rate': {
               'threshold': 0.95,  # 成功率低于95%
               'severity': 'high',
               'description': '路由成功率过低'
           }
       }
       
       def __init__(self):
           self.alert_center = get_alert_center()
           self._monitoring = False
       
       async def start_monitoring(self):
           """启动监控"""
           self._monitoring = True
           while self._monitoring:
               await self._check_routing_health()
               await asyncio.sleep(30)  # 30秒检查一次
       
       async def _check_routing_health(self):
           """检查路由健康状态"""
           stats = get_routing_stats()
           
           # 检查失败率
           if stats['failure_rate'] > self.ALERT_RULES['routing_failure']['threshold']:
               self._create_alert('routing_failure', stats)
           
           # 检查延迟
           if stats['avg_latency'] > self.ALERT_RULES['high_latency']['threshold']:
               self._create_alert('high_latency', stats)
           
           # 检查成功率
           if stats['success_rate'] < self.ALERT_RULES['low_success_rate']['threshold']:
               self._create_alert('low_success_rate', stats)
       
       def _create_alert(self, rule_id: str, data: dict):
           """创建告警"""
           rule = self.ALERT_RULES[rule_id]
           self.alert_center.create_alert(
               title=f"路由监控告警: {rule['description']}",
               message=f"当前值: {data}",
               severity=rule['severity'],
               source="OrderRoutingMonitor"
           )
   ```

2. **前端告警显示**

1. **路由追溯接口**
   ```python
   @router.get("/trading/routing/trace/{trace_id}")
   async def get_routing_trace(trace_id: str) -> Dict[str, Any]:
       """
       根据trace_id查询完整路由链
       
       返回完整的路由决策链，包括：
       - 原始订单信息
       - 路由决策过程
       - 执行结果
       - 时间戳链
       """
       try:
           # 从持久化存储查询
           from .routing_persistence import get_routing_persistence
           persistence = get_routing_persistence()
           
           trace_data = persistence.get_trace_by_id(trace_id)
           
           if not trace_data:
               raise HTTPException(status_code=404, detail="路由链不存在")
           
           return {
               "trace_id": trace_id,
               "chain": trace_data,
               "timestamp": int(time.time())
           }
       except HTTPException:
           raise
       except Exception as e:
           logger.error(f"查询路由链失败: {e}")
           raise HTTPException(status_code=500, detail=f"查询失败: {str(e)}")
   
   
   @router.get("/trading/routing/traces")
   async def get_routing_traces(
       start_time: Optional[str] = None,
       end_time: Optional[str] = None,
       strategy_id: Optional[str] = None,
       limit: int = 100
   ) -> Dict[str, Any]:
       """查询路由链列表"""
       try:
           from .routing_persistence import get_routing_persistence
           persistence = get_routing_persistence()
           
           traces = persistence.query_traces(
               start_time=start_time,
               end_time=end_time,
               strategy_id=strategy_id,
               limit=limit
           )
           
           return {
               "traces": traces,
               "total": len(traces),
               "timestamp": int(time.time())
           }
       except Exception as e:
           logger.error(f"查询路由链列表失败: {e}")
           raise HTTPException(status_code=500, detail=f"查询失败: {str(e)}")
   ```

2. **前端路由链可视化**
