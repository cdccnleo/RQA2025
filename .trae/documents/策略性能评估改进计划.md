```javascript
function maskSensitiveParams(params) {
    if (!params) return params;
    const masked = {...params};
    const sensitiveFields = ['api_key', 'secret', 'password', 'token', 'private_key'];
    sensitiveFields.forEach(field => {
        if (masked[field]) masked[field] = '***';
    });
    return masked;
}
```

```python
def validate_metrics(metrics: dict) -> tuple:
    """验证指标计算结果合理性"""
    warnings = []
    
    # 检查夏普比率
    sharpe = metrics.get('sharpe_ratio', 0)
    if sharpe > 10:
        warnings.append(f"夏普比率异常: {sharpe:.2f}（超过10）")
    elif sharpe < -5:
        warnings.append(f"夏普比率异常: {sharpe:.2f}（低于-5）")
    
    # 检查最大回撤
    max_dd = metrics.get('max_drawdown', 0)
    if max_dd > 0.99:
        warnings.append(f"最大回撤异常: {max_dd:.2%}（超过99%）")
    
    # 检查年化收益
    annual_return = metrics.get('annual_return', 0)
    if annual_return > 10:  # 超过1000%
        warnings.append(f"年化收益异常: {annual_return:.2%}（超过1000%）")
    
    return len(warnings) == 0, warnings
```

```javascript
// 检查指标是否异常
function isMetricAbnormal(metric, value) {
    const thresholds = {
        'sharpe_ratio': { min: -5, max: 10 },
        'max_drawdown': { min: 0, max: 0.99 },
        'annual_return': { min: -1, max: 10 }
    };
    
    if (thresholds[metric]) {
        return value < thresholds[metric].min || value > thresholds[metric].max;
    }
    return false;
}

// 渲染时标识异常
function renderMetricCell(metric, value) {
    const isAbnormal = isMetricAbnormal(metric, value);
    const className = isAbnormal ? 'bg-red-100 text-red-800' : '';
    const title = isAbnormal ? '数据异常，请检查' : '';
    return `<td class="${className}" title="${title}">${value}</td>`;
}
```
