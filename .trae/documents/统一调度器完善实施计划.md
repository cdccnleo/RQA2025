   async def should_retry_task(self, task_id: str) -> bool:
       """
       检查任务是否应该重试
       
       Args:
           task_id: 任务ID
       
       Returns:
           bool: 是否应该重试
       """
       task = self.get_task(task_id)
       if not task:
           return False
       
       # 检查是否还有重试次数
       if task.retry_count >= task.max_retries:
           return False
       
       # 检查任务类型是否支持重试
       # 某些任务类型可能不支持重试（如订单执行）
       non_retryable_types = ["order_execution", "trade_processing"]
       if task.type in non_retryable_types:
           return False
       
       return True
   
   async def retry_task(self, task_id: str) -> Optional[str]:
       """
       重试任务
       
       Args:
           task_id: 原任务ID
       
       Returns:
           Optional[str]: 新任务ID，如果不需要重试则返回None
       """
       old_task = self.get_task(task_id)
       if not old_task:
           return None
       
       # 检查是否应该重试
       if not await self.should_retry_task(task_id):
           return None
       
       # 创建新任务
       new_task_id = await self.create_task(
           task_type=old_task.type,
           payload=old_task.payload,
           priority=old_task.priority,
           timeout_seconds=old_task.timeout_seconds,
           max_retries=old_task.max_retries,
           retry_delay=old_task.retry_delay * 2  # 指数退避
       )
       
       # 更新重试计数
       old_task.retry_count += 1
       
       print(f"任务 {task_id} 重试，新任务ID: {new_task_id} "
             f"(重试次数: {old_task.retry_count}/{old_task.max_retries})")
       
       return new_task_id
   ```

   from sqlalchemy import create_engine

   from sqlalchemy.exc import SQLAlchemyError
   import os
   
   from .models import Base, TaskModel, JobModel
   from .base import Task, TaskStatus, Job
   
   class PersistenceManager:
       """持久化管理器"""
       
       def __init__(self, database_url: Optional[str] = None):
           """
           初始化持久化管理器
           
           Args:
               database_url: 数据库连接URL，默认使用环境变量
           """
           if database_url is None:
               database_url = os.getenv(
                   "SCHEDULER_DATABASE_URL",
                   "postgresql://localhost:5432/rqa2025"
               )
           
           self.engine = create_engine(database_url)
           self.SessionLocal = sessionmaker(bind=self.engine)
           
           # 创建表
           Base.metadata.create_all(self.engine)
       
       def save_task(self, task: Task) -> bool:
           """保存任务到数据库"""
           session: Session = self.SessionLocal()
           try:
               task_model = TaskModel(
                   id=task.id,
                   type=task.type,
                   status=task.status.value,
                   priority=task.priority,
                   created_at=task.created_at,
                   started_at=task.started_at,
                   completed_at=task.completed_at,
                   payload=task.payload,
                   result=task.result,
                   error=task.error,
                   worker_id=task.worker_id,
                   timeout_seconds=task.timeout_seconds,
                   max_retries=task.max_retries,
                   retry_count=task.retry_count,
                   retry_delay=task.retry_delay,
                   deadline=task.deadline
               )
               
               session.merge(task_model)
               session.commit()
               return True
           except SQLAlchemyError as e:
               print(f"保存任务失败: {e}")
               session.rollback()
               return False
           finally:
               session.close()
       
       def get_task(self, task_id: str) -> Optional[Task]:
           """从数据库获取任务"""
           session: Session = self.SessionLocal()
           try:
               model = session.query(TaskModel).filter(TaskModel.id == task_id).first()
               if model:
                   return self._model_to_task(model)
               return None
           finally:
               session.close()
       
       def get_tasks_by_status(self, status: TaskStatus) -> List[Task]:
           """获取指定状态的任务"""
           session: Session = self.SessionLocal()
           try:

               ).all()
               return [self._model_to_task(m) for m in models]

   # event_integration.py
   from typing import Dict, Any, Callable, List
   from enum import Enum
   from datetime import datetime
   import asyncio
   
   class SchedulerEventType(Enum):
       """调度器事件类型"""
       TASK_SUBMITTED = "task.submitted"
       TASK_STARTED = "task.started"
       TASK_COMPLETED = "task.completed"
       TASK_FAILED = "task.failed"
       TASK_CANCELLED = "task.cancelled"
       TASK_TIMEOUT = "task.timeout"
       TASK_RETRY = "task.retry"
       
       JOB_TRIGGERED = "job.triggered"
       JOB_ENABLED = "job.enabled"
       JOB_DISABLED = "job.disabled"
       
       WORKER_STARTED = "worker.started"
       WORKER_STOPPED = "worker.stopped"
       WORKER_HEARTBEAT = "worker.heartbeat"
       
       SCHEDULER_STARTED = "scheduler.started"
       SCHEDULER_STOPPED = "scheduler.stopped"
   
   class EventBusIntegration:
       """事件总线集成"""
       
       def __init__(self):
           self._event_bus = None
           self._local_handlers: Dict[SchedulerEventType, List[Callable]] = {}
       
       def initialize(self, event_bus=None):
           """初始化事件总线连接"""
           if event_bus:
               self._event_bus = event_bus
           else:
               # 尝试从核心服务层导入
               try:
                   from src.core.event_bus.core import EventBus
                   self._event_bus = EventBus()
               except ImportError:
                   print("警告: 无法导入核心事件总线，使用本地事件处理")
       
       async def emit(self, event_type: SchedulerEventType, data: Dict[str, Any]):
           """发布事件"""
           event = {
               "type": event_type.value,
               "timestamp": datetime.now().isoformat(),
               "data": data
           }
           
           # 发布到全局事件总线
           if self._event_bus:
               try:
                   await self._event_bus.publish(event_type.value, event)
               except Exception as e:
                   print(f"发布事件到全局总线失败: {e}")
           
           # 调用本地处理器
           handlers = self._local_handlers.get(event_type, [])
           for handler in handlers:
               try:
                   if asyncio.iscoroutinefunction(handler):
                       await handler(event)
                   else:
                       handler(event)
               except Exception as e:
                   print(f"事件处理器错误: {e}")
       
       def on(self, event_type: SchedulerEventType, handler: Callable):
           """订阅事件"""
           if event_type not in self._local_handlers:
               self._local_handlers[event_type] = []
           self._local_handlers[event_type].append(handler)
       
       def off(self, event_type: SchedulerEventType, handler: Callable):
           """取消订阅"""
           if event_type in self._local_handlers:
               if handler in self._local_handlers[event_type]:
                   self._local_handlers[event_type].remove(handler)
   ```

   ```python
   # priority_queue.py
   import heapq
   from typing import Any, Tuple, List
   from queue import Empty
   import threading
   
   class PriorityTaskQueue:
       """优先级任务队列"""
       
       def __init__(self):
           self._queue: List[Tuple[int, int, Any]] = []  # (priority, seq, item)
           self._count = 0
           self._lock = threading.Lock()
           self._not_empty = threading.Condition(self._lock)
       
       def put(self, item: Any, priority: int = 5, block: bool = True, timeout: float = None):
           """
           添加任务到队列
           
           Args:
               item: 任务数据
               priority: 优先级（数字越小优先级越高）
               block: 是否阻塞
               timeout: 超时时间
           """
           with self._lock:
               heapq.heappush(self._queue, (priority, self._count, item))
               self._count += 1
               self._not_empty.notify()
       
       def get(self, block: bool = True, timeout: float = None) -> Any:
           """获取任务"""
           with self._not_empty:
               if not block:
                   if not self._queue:
                       raise Empty()
               elif timeout is None:
                   while not self._queue:
                       self._not_empty.wait()
               elif timeout < 0:
                   raise ValueError("timeout must be non-negative")
               else:
                   endtime = time.time() + timeout
                   while not self._queue:
                       remaining = endtime - time.time()
                       if remaining <= 0:
                           raise Empty()
                       self._not_empty.wait(remaining)
               
               return heapq.heappop(self._queue)[2]
       
       def qsize(self) -> int:
           """获取队列大小"""
           with self._lock:
               return len(self._queue)
       
       def empty(self) -> bool:
           """检查队列是否为空"""
           with self._lock:
               return not self._queue
   ```
