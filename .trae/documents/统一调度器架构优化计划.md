from typing import Dict, Any, List, Optional

class TaskStatus(Enum):
    PENDING = "pending"
    RUNNING = "running"
    PAUSED = "paused"
    COMPLETED = "completed"
    FAILED = "failed"
    CANCELLED = "cancelled"

    FEATURE_EXTRACTION = "feature_extraction"
    MODEL_TRAINING = "model_training"
    MODEL_INFERENCE = "model_inference"

@dataclass
class Task:
    id: str
    type: str
    status: TaskStatus
    priority: int
    created_at: datetime
    started_at: Optional[datetime] = None
    completed_at: Optional[datetime] = None
    payload: Dict[str, Any] = None
    result: Any = None
    error: Optional[str] = None

class BaseScheduler(ABC):
    """调度器抽象基类"""

        pass

```python
"""任务管理器 - 负责任务的CRUD和状态管理"""

import asyncio
from typing import Dict, List, Optional
from datetime import datetime, timedelta
import json

class TaskManager:
    """任务管理器"""
    
    def __init__(self, max_history: int = 1000):
        self._tasks: Dict[str, Task] = {}
        self._task_history: List[Task] = []
        self._max_history = max_history
        self._lock = asyncio.Lock()
    
    async def create_task(
        self,
        task_type: str,
        payload: Dict,
        priority: int = 5
    ) -> str:
        """创建新任务"""
        async with self._lock:
            task_id = self._generate_task_id()
            task = Task(
                id=task_id,
                type=task_type,
                status=TaskStatus.PENDING,
                priority=priority,
                created_at=datetime.now(),
                payload=payload
            )
            self._tasks[task_id] = task
            return task_id
    
    async def update_task_status(
        self,
        task_id: str,
        status: TaskStatus,
        result: Any = None,
        error: str = None
    ):
        """更新任务状态"""
        async with self._lock:
            if task_id not in self._tasks:
                return
            
            task = self._tasks[task_id]
            task.status = status
            
            if status == TaskStatus.RUNNING:
                task.started_at = datetime.now()
            elif status in [TaskStatus.COMPLETED, TaskStatus.FAILED, TaskStatus.CANCELLED]:
                task.completed_at = datetime.now()
                task.result = result
                task.error = error
                
                # 移动到历史记录
                self._task_history.append(task)
                del self._tasks[task_id]
                
                # 限制历史记录大小
                if len(self._task_history) > self._max_history:
                    self._task_history = self._task_history[-self._max_history:]
    
    def get_task(self, task_id: str) -> Optional[Task]:
        """获取任务"""
        # 先在活跃任务中查找
        if task_id in self._tasks:
            return self._tasks[task_id]
        
        # 再在历史记录中查找
        for task in self._task_history:
            if task.id == task_id:
                return task
        
        return None
    
    def get_running_tasks(self) -> List[Task]:
        """获取运行中任务"""
        return [
            task for task in self._tasks.values()
            if task.status == TaskStatus.RUNNING
        ]
    
    def get_pending_tasks(self) -> List[Task]:
        """获取等待中任务"""
        return [
            task for task in self._tasks.values()
            if task.status == TaskStatus.PENDING
        ]
    
    def get_completed_tasks(
        self,
        limit: int = 20,
        offset: int = 0
    ) -> List[Task]:
        """获取已完成任务"""
        sorted_history = sorted(
            self._task_history,
            key=lambda t: t.completed_at or datetime.min,
            reverse=True
        )
        return sorted_history[offset:offset + limit]
    
    def get_statistics(self) -> Dict:
        """获取统计信息"""
        total = len(self._tasks) + len(self._task_history)
        running = len(self.get_running_tasks())
        pending = len(self.get_pending_tasks())
        completed = len([t for t in self._task_history if t.status == TaskStatus.COMPLETED])
        failed = len([t for t in self._task_history if t.status == TaskStatus.FAILED])
        
        return {
            "total": total,
            "running": running,
            "pending": pending,
            "completed": completed,
            "failed": failed,
            "success_rate": completed / (completed + failed) if (completed + failed) > 0 else 0
        }
    
    def _generate_task_id(self) -> str:
        """生成任务ID"""
        import uuid
        return f"task-{uuid.uuid4().hex[:8]}"
```

```python
"""工作进程管理器"""

import asyncio
from typing import Dict, List, Optional, Callable
from dataclasses import dataclass
from datetime import datetime
import threading

@dataclass
class Worker:
    id: str
    status: str  # "idle", "busy", "stopped"
    current_task: Optional[str] = None
    started_at: datetime = None
    last_heartbeat: datetime = None
    thread: Optional[threading.Thread] = None

class WorkerManager:
    """工作进程管理器"""
    
    def __init__(self, max_workers: int = 4):
        self._max_workers = max_workers
        self._workers: Dict[str, Worker] = {}
        self._task_queue = asyncio.Queue()
        self._running = False
        self._lock = asyncio.Lock()
    
    async def start(self):
        """启动工作进程池"""
        self._running = True
        
        # 创建工作进程
        for i in range(self._max_workers):
            worker_id = f"worker-{i+1}"
            worker = Worker(
                id=worker_id,
                status="idle",
                started_at=datetime.now(),
                last_heartbeat=datetime.now()
            )
            self._workers[worker_id] = worker
            
            # 启动工作线程
            thread = threading.Thread(
                target=self._worker_loop,
                args=(worker_id,),
                daemon=True
            )
            worker.thread = thread
            thread.start()
    
    async def stop(self):
        """停止工作进程池"""
        self._running = False
        
        # 等待所有工作线程结束
        for worker in self._workers.values():
            if worker.thread and worker.thread.is_alive():
                worker.thread.join(timeout=5)
    
    def _worker_loop(self, worker_id: str):
        """工作线程主循环"""
        worker = self._workers[worker_id]
        
        while self._running:
            try:
                # 更新心跳
                worker.last_heartbeat = datetime.now()
                
                # 获取任务（非阻塞）
                try:
                    task = self._task_queue.get_nowait()
                except asyncio.QueueEmpty:
                    asyncio.sleep(0.1)
                    continue
                
                # 执行任务
                worker.status = "busy"
                worker.current_task = task.get("id")
                
                # 调用任务处理器
                handler = task.get("handler")
                if handler:
                    try:
                        result = handler(task.get("payload"))
                        task["result"] = result
                        task["status"] = "completed"
                    except Exception as e:
                        task["error"] = str(e)
                        task["status"] = "failed"
                
                worker.status = "idle"
                worker.current_task = None
                
            except Exception as e:
                print(f"Worker {worker_id} error: {e}")
    
    def get_workers(self) -> List[Worker]:
        """获取所有工作进程"""
        return list(self._workers.values())
    
    def get_active_workers(self) -> List[Worker]:
        """获取活跃工作进程"""
        return [
            w for w in self._workers.values()
            if w.status == "busy"
        ]
    
    def get_statistics(self) -> Dict:
        """获取统计信息"""
        total = len(self._workers)
        active = len(self.get_active_workers())
        idle = total - active
        
        return {
            "total": total,
            "active": active,
            "idle": idle
        }
```
