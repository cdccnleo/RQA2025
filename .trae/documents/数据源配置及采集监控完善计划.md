#### 7.1 健康检测算法
```python
async def check_datasource_health(source_config: Dict) -> HealthStatus:
    """检测数据源健康状态"""
    start_time = time.time()
    
    try:
        # 测试连接
        response = await test_connection(source_config)
        
        response_time = (time.time() - start_time) * 1000
        
        if response.status == 200:
            return HealthStatus(
                status='healthy',
                response_time_ms=response_time,
                message='连接正常'
            )
        else:
            return HealthStatus(
                status='unhealthy',
                response_time_ms=response_time,
                message=f'HTTP {response.status}'
            )
            
    except asyncio.TimeoutError:
        return HealthStatus(
            status='timeout',
            response_time_ms=10000,
            message='连接超时'
        )
    except Exception as e:
        return HealthStatus(
            status='error',
            response_time_ms=0,
            message=str(e)
        )
```

#### 7.2 告警规则引擎
```python
class AlertRuleEngine:
    """告警规则引擎"""
    
    RULES = {
        'connection_failure': {
            'condition': 'consecutive_failures >= 3',
            'severity': 'warning',
            'cooldown_minutes': 30
        },
        'response_timeout': {
            'condition': 'response_time_ms > 10000',
            'severity': 'warning',
            'cooldown_minutes': 15
        },
        'unavailable': {
            'condition': 'unavailable_duration_minutes >= 5',
            'severity': 'critical',
            'cooldown_minutes': 60
        }
    }
    
    def evaluate(self, health_status: HealthStatus) -> Optional[Alert]:
        """评估告警规则"""
        for rule_name, rule_config in self.RULES.items():
            if self._check_condition(rule_config['condition'], health_status):
                return Alert(
                    rule=rule_name,
                    severity=rule_config['severity'],
                    message=self._generate_message(rule_name, health_status)
                )
        return None
```
