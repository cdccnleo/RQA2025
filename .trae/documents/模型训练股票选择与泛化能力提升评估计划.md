```python
def filter_stocks_by_quality(df: pd.DataFrame, min_volume: float = 1e7,
                             min_data_points: int = 500) -> bool:
    """筛选高质量股票"""
    # 检查数据完整率
    completeness = df.notna().mean().mean()
    if completeness < 0.95:
        return False
    
    # 检查平均成交量
    avg_volume = df['volume'].mean()
    if avg_volume < min_volume:
        return False
    
    # 检查数据点数量
    if len(df) < min_data_points:
        return False
    
    return True
```

### 2.3 波动性分层策略

**目标**: 选择不同波动性的股票，增加样本多样性

**实现方案**:
```python
def select_stocks_by_volatility(symbols_data: Dict[str, pd.DataFrame],
                                n_low: int = 3, n_mid: int = 3, n_high: int = 3) -> List[str]:
    """按波动性分层选择股票"""
    volatility_scores = {}
    
    for symbol, df in symbols_data.items():
        returns = df['close'].pct_change().dropna()
        volatility = returns.std() * np.sqrt(252)  # 年化波动率
        volatility_scores[symbol] = volatility
    
    # 按波动性排序
    sorted_stocks = sorted(volatility_scores.items(), key=lambda x: x[1])
    
    # 分层选择
    n = len(sorted_stocks)
    low_vol = [s for s, _ in sorted_stocks[:n//3]]
    mid_vol = [s for s, _ in sorted_stocks[n//3:2*n//3]]
    high_vol = [s for s, _ in sorted_stocks[2*n//3:]]
    
    selected = (low_vol[:n_low] + mid_vol[:n_mid] + high_vol[:n_high])
    return selected
```

```python
class StockRecommender:
    """股票推荐系统"""
    
    def __init__(self):
        self.sector_weights = {
            '银行': 0.15,
            '电子': 0.20,
            '通信': 0.15,
            '金融': 0.10,
            '新能源': 0.20,
            '传媒': 0.10,
            '其他': 0.10
        }
    
    def recommend_stocks(self, available_symbols: List[str], n_stocks: int = 10) -> List[str]:
        """推荐股票组合"""
        # 按行业权重选择
        selected = []
        sector_count = {}
        
        for symbol in available_symbols:
            sector = self._get_sector(symbol)
            target_count = int(n_stocks * self.sector_weights.get(sector, 0.1))
            
            if sector_count.get(sector, 0) < target_count:
                selected.append(symbol)
                sector_count[sector] = sector_count.get(sector, 0) + 1
            
            if len(selected) >= n_stocks:
                break
        
        return selected
```

### 3.2 时间序列交叉验证

**实现方案**:
```python
from sklearn.model_selection import TimeSeriesSplit

def time_series_cross_validation(model, X, y, n_splits=5):
    """时间序列交叉验证"""
    tscv = TimeSeriesSplit(n_splits=n_splits)
    scores = []
    
    for train_idx, test_idx in tscv.split(X):
        X_train, X_test = X.iloc[train_idx], X.iloc[test_idx]
        y_train, y_test = y.iloc[train_idx], y.iloc[test_idx]
        
        model.fit(X_train, y_train)
        score = model.score(X_test, y_test)
        scores.append(score)
    
    return np.mean(scores), np.std(scores)
```

### 3.3  walk-forward验证

**实现方案**:
```python
def walk_forward_validation(model, X, y, initial_train_size=0.6, step=0.05):
    """Walk-forward验证"""
    n = len(X)
    train_end = int(n * initial_train_size)
    step_size = int(n * step)
    
    scores = []
    
    while train_end < n - step_size:
        X_train = X.iloc[:train_end]
        y_train = y.iloc[:train_end]
        X_test = X.iloc[train_end:train_end + step_size]
        y_test = y.iloc[train_end:train_end + step_size]
        
        model.fit(X_train, y_train)
        score = model.score(X_test, y_test)
        scores.append(score)
        
        train_end += step_size
    
    return scores
```

```python
class EnsembleModel:
    """集成模型"""
    
    def __init__(self, models: List[Any], weights: Optional[List[float]] = None):
        self.models = models
        self.weights = weights or [1.0 / len(models)] * len(models)
    
    def predict(self, X: pd.DataFrame) -> np.ndarray:
        """集成预测"""
        predictions = []
        for model, weight in zip(self.models, self.weights):
            pred = model.predict(X) * weight
            predictions.append(pred)
        
        return np.sum(predictions, axis=0)
    
    @staticmethod
    def create_diverse_models(config: Dict[str, Any]) -> List[Any]:
        """创建多样化的模型集合"""
        models = []
        
        # 不同类型的模型
        from sklearn.ensemble import RandomForestClassifier
        from xgboost import XGBClassifier
        
        models.append(RandomForestClassifier(n_estimators=100, max_depth=10))
        models.append(RandomForestClassifier(n_estimators=200, max_depth=15))
        models.append(XGBClassifier(n_estimators=100, max_depth=6))
        
        return models
```

```python
def analyze_feature_importance(model, feature_names: List[str]) -> pd.DataFrame:
    """分析特征重要性"""
    if hasattr(model, 'feature_importances_'):
        importances = model.feature_importances_
    elif hasattr(model, 'coef_'):
        importances = np.abs(model.coef_[0])
    else:
        return None
    
    importance_df = pd.DataFrame({
        'feature': feature_names,
        'importance': importances
    }).sort_values('importance', ascending=False)
    
    return importance_df
```
