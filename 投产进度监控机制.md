## 🎯 监控机制概述

### 监控目的
1. **实时跟踪**: 及时了解项目进度和质量状态
2. **风险预警**: 提前发现问题和风险
3. **数据驱动**: 基于数据做决策和调整
4. **持续改进**: 识别改进机会，优化流程

### 监控原则
- **自动化优先**: 尽量使用自动化工具
- **实时更新**: 关键指标实时或每日更新
- **可视化展示**: 使用图表直观展示
- **异常告警**: 关键指标超阈值自动告警

#### 1. 每日测试监控脚本
```bash
#!/bin/bash
# daily_test_monitor.sh
# 每日自动运行测试并生成报告

DATE=$(date +%Y%m%d)
LOG_DIR="test_logs"

echo "=== RQA2025 每日测试监控 - $DATE ==="

# 1. 运行测试收集
echo "1. 收集测试..."
pytest tests/ --co -q > $LOG_DIR/collection_$DATE.txt 2>&1

# 2. 运行Infrastructure/utils测试
echo "2. 运行Infrastructure/utils测试..."
pytest tests/unit/infrastructure/utils/ -v \
  --cov=src/infrastructure/utils \
  --cov-report=html \
  --cov-report=term \
  > $LOG_DIR/infra_utils_$DATE.txt 2>&1

# 3. 统计数据
echo "3. 统计测试数据..."
PASSED=$(grep -c "PASSED" $LOG_DIR/infra_utils_$DATE.txt || echo 0)
FAILED=$(grep -c "FAILED" $LOG_DIR/infra_utils_$DATE.txt || echo 0)
ERRORS=$(grep -c "ERROR" $LOG_DIR/collection_$DATE.txt || echo 0)
COVERAGE=$(grep "TOTAL" $LOG_DIR/infra_utils_$DATE.txt | awk '{print $NF}')

# 4. 生成简报
cat > $LOG_DIR/daily_report_$DATE.txt <<EOF
=== RQA2025 每日测试报告 - $DATE ===

测试执行情况:
- 通过数: $PASSED
- 失败数: $FAILED
- 收集错误: $ERRORS
- 覆盖率: $COVERAGE

对比昨日:
- [需要计算差异]

状态评估:
- $([ $ERRORS -eq 0 ] && echo "✅ 收集正常" || echo "🔴 收集有错误")
- $([ $FAILED -lt 50 ] && echo "✅ 失败数可接受" || echo "🔴 失败数过多")
- 覆盖率: $(echo $COVERAGE | grep -q "45\." && echo "🟢 正常" || echo "🔴 异常")

EOF

echo "=== 监控完成，报告已生成: $LOG_DIR/daily_report_$DATE.txt ==="
```

#### 2. 覆盖率趋势监控脚本
```python
# coverage_trend_monitor.py
# 监控覆盖率趋势

import json
import os
from datetime import datetime, timedelta
import matplotlib.pyplot as plt

def collect_coverage_data():
    """收集覆盖率数据"""
    # 运行pytest-cov获取覆盖率
    os.system("pytest tests/unit/infrastructure/utils/ "
              "--cov=src/infrastructure/utils "
              "--cov-report=json "
              "> /dev/null 2>&1")
    
    # 读取coverage.json
    with open('.coverage.json', 'r') as f:
        data = json.load(f)
    
    coverage = data['totals']['percent_covered']
    
    # 保存到历史数据
    today = datetime.now().strftime('%Y-%m-%d')
    history_file = 'test_logs/coverage_history.json'
    
    if os.path.exists(history_file):
        with open(history_file, 'r') as f:
            history = json.load(f)
    else:
        history = {}
    
    history[today] = {
        'coverage': coverage,
        'timestamp': datetime.now().isoformat()
    }
    
    with open(history_file, 'w') as f:
        json.dump(history, f, indent=2)
    
    return coverage, history

def plot_coverage_trend(history):
    """绘制覆盖率趋势图"""
    dates = sorted(history.keys())
    coverages = [history[d]['coverage'] for d in dates]
    
    plt.figure(figsize=(12, 6))
    plt.plot(dates, coverages, marker='o', linewidth=2)
    plt.axhline(y=52, color='g', linestyle='--', label='Week 2 目标 (52%)')
    plt.axhline(y=80, color='r', linestyle='--', label='Week 12 目标 (80%)')
    plt.title('测试覆盖率趋势')
    plt.xlabel('日期')
    plt.ylabel('覆盖率 (%)')
    plt.legend()
    plt.grid(True)
    plt.xticks(rotation=45)
    plt.tight_layout()
    plt.savefig('test_logs/coverage_trend.png')
    print("✅ 趋势图已生成: test_logs/coverage_trend.png")

def check_coverage_alert(coverage, history):
    """检查覆盖率告警"""
    if len(history) < 2:
        return
    
    dates = sorted(history.keys())
    yesterday = dates[-2]
    yesterday_coverage = history[yesterday]['coverage']
    
    diff = coverage - yesterday_coverage
    
    if diff < -1:
        print(f"🔴 告警: 覆盖率下降 {abs(diff):.2f}% "
              f"(昨日: {yesterday_coverage:.2f}%, 今日: {coverage:.2f}%)")
    elif diff > 0:
        print(f"✅ 覆盖率提升 {diff:.2f}% "
              f"(昨日: {yesterday_coverage:.2f}%, 今日: {coverage:.2f}%)")
    else:
        print(f"➡️ 覆盖率保持 {coverage:.2f}%")

if __name__ == '__main__':
    print("=== 覆盖率趋势监控 ===")
    coverage, history = collect_coverage_data()
    print(f"当前覆盖率: {coverage:.2f}%")
    check_coverage_alert(coverage, history)
    plot_coverage_trend(history)
```

#### 3. 测试通过率监控脚本
```python
# test_pass_rate_monitor.py
# 监控测试通过率

import subprocess
import json
import os
from datetime import datetime

def run_tests_and_collect_stats():
    """运行测试并收集统计"""
    result = subprocess.run(
        ['pytest', 'tests/unit/infrastructure/utils/', '-v', '-q'],
        capture_output=True,
        text=True
    )
    
    output = result.stdout + result.stderr
    
    # 解析统计
    passed = output.count(' PASSED')
    failed = output.count(' FAILED')
    errors = output.count(' ERROR')
    total = passed + failed
    pass_rate = (passed / total * 100) if total > 0 else 0
    
    return {
        'passed': passed,
        'failed': failed,
        'errors': errors,
        'total': total,
        'pass_rate': pass_rate,
        'timestamp': datetime.now().isoformat()
    }

def save_stats(stats):
    """保存统计数据"""
    history_file = 'test_logs/pass_rate_history.json'
    
    if os.path.exists(history_file):
        with open(history_file, 'r') as f:
            history = json.load(f)
    else:
        history = []
    
    history.append(stats)
    
    with open(history_file, 'w') as f:
        json.dump(history, f, indent=2)

def check_pass_rate_alert(stats, history):
    """检查通过率告警"""
    if len(history) < 2:
        return
    
    yesterday_rate = history[-2]['pass_rate']
    today_rate = stats['pass_rate']
    diff = today_rate - yesterday_rate
    
    print(f"\n当前通过率: {today_rate:.1f}%")
    print(f"通过: {stats['passed']}, 失败: {stats['failed']}, 错误: {stats['errors']}")
    
    if diff < -2:
        print(f"🔴 告警: 通过率下降 {abs(diff):.1f}% "
              f"(昨日: {yesterday_rate:.1f}%)")
    elif diff > 2:
        print(f"✅ 通过率提升 {diff:.1f}% "
              f"(昨日: {yesterday_rate:.1f}%)")
    else:
        print(f"➡️ 通过率保持稳定")
    
    # Week 1目标检查
    if stats['pass_rate'] < 72:
        print(f"⚠️ 提示: 距离Week 1目标(72%)还差 {72 - stats['pass_rate']:.1f}%")

if __name__ == '__main__':
    print("=== 测试通过率监控 ===")
    stats = run_tests_and_collect_stats()
    
    # 加载历史数据
    history_file = 'test_logs/pass_rate_history.json'
    if os.path.exists(history_file):
        with open(history_file, 'r') as f:
            history = json.load(f)
    else:
        history = []
    
    check_pass_rate_alert(stats, history)
    save_stats(stats)
```

## 🚨 告警机制

### 告警规则

#### 一级告警（立即处理）
```python
告警条件:
1. 测试收集错误新增 >5个
2. 测试通过率下降 >5%
3. 覆盖率下降 >2%
4. P0问题新增

告警方式:
- 邮件通知项目经理和架构师
- 项目群@all通知
- 站会上重点讨论

响应时间: 2小时内
```

#### 二级告警（当日处理）
```python
告警条件:
1. 测试通过率下降 2-5%
2. 覆盖率下降 1-2%
3. 测试执行时间 >20分钟
4. P1问题新增

告警方式:
- 项目群通知
- 站会上讨论

响应时间: 当日内
```

#### 三级告警（本周处理）
```python
告警条件:
1. 任务延期 >1天
2. 代码质量指标下降
3. 构建成功率 <95%

告警方式:
- 记录到问题跟踪表
- 周会上讨论

响应时间: 本周内
```

### 告警处理流程
```
1. 告警触发
   ↓
2. 自动通知相关人员
   ↓
3. 分析告警原因
   ↓
4. 制定解决方案
   ↓
5. 执行解决措施
   ↓
6. 验证问题解决
   ↓
7. 更新文档和总结
```
