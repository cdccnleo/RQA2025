# RQA2025 Logger环境变量配置示例
# 复制此文件的内容到你的环境变量配置中

# 日志基本配置
LOGGER_LEVEL=INFO
LOGGER_FORMAT=STRUCTURED
LOGGER_CATEGORY=GENERAL
LOGGER_LOG_DIR=logs

# 异步配置
LOGGER_ASYNC_ENABLED=true
LOGGER_ASYNC_QUEUE_SIZE=1000
LOGGER_FLUSH_INTERVAL=1.0

# 对象池配置
LOGGER_POOL_ENABLED=true
LOGGER_POOL_MAX_SIZE=100
LOGGER_POOL_IDLE_TIMEOUT=300

# 文件处理器配置
LOGGER_MAX_FILE_SIZE=10485760
LOGGER_BACKUP_COUNT=5

# 专用Logger配置
LOGGER_BUSINESS_LEVEL=INFO
LOGGER_AUDIT_LEVEL=WARNING
LOGGER_TRADING_LEVEL=INFO
LOGGER_RISK_LEVEL=WARNING
LOGGER_PERFORMANCE_LEVEL=INFO
LOGGER_DATABASE_LEVEL=INFO

# 全局配置
LOGGER_ENCODING=utf-8
LOGGER_TIMEZONE=UTC
LOGGER_MAX_LOG_AGE=30
LOGGER_COMPRESSION=gzip

# 监控和告警配置 (JSON格式)
LOGGER_MONITORING_CONFIG={"enabled": true, "metrics_prefix": "logger", "push_interval": 60}
LOGGER_ALERT_CONFIG={"enabled": true, "error_threshold": 10, "channels": ["email", "slack"]}

# 安全配置
LOGGER_ENABLE_ENCRYPTION=false
LOGGER_AUDIT_TRAIL=true
