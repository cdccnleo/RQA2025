##### 架构设计题目
```python
"""
题目: 设计RQA2026高并发量化交易系统的整体架构

要求:
1. 支持每秒10万笔交易处理
2. 保证交易顺序和数据一致性
3. 实现实时风控和监控
4. 支持水平扩展和故障转移
5. 考虑成本效益和运维复杂度

请提供:
- 整体架构图和组件说明
- 关键技术选型和理由
- 性能优化策略和扩展方案
- 监控告警和故障处理机制
- 成本估算和ROI分析
"""

# 期望回答要点
class TradingSystemArchitecture:
    def __init__(self):
        self.order_gateway = OrderGateway()      # 订单入口
        self.matching_engine = MatchingEngine()  # 撮合引擎
        self.risk_engine = RiskEngine()          # 风控引擎
        self.execution_engine = ExecutionEngine() # 执行引擎
        self.monitoring_system = MonitoringSystem() # 监控系统

    async def process_order(self, order: Order):
        # 订单预处理和验证
        validated_order = await self.validate_order(order)

        # 风险检查
        risk_result = await self.risk_engine.check_risk(validated_order)
        if not risk_result.approved:
            return self.reject_order(order, risk_result.reason)

        # 订单撮合和执行
        match_result = await self.matching_engine.match_order(validated_order)
        execution_result = await self.execution_engine.execute_match(match_result)

        # 结果记录和监控
        await self.monitoring_system.record_trade(execution_result)
        return execution_result
```

### 综合评分算法
```python
class InterviewEvaluator:
    def __init__(self):
        self.dimensions = {
            'strategic_thinking': 0.25,
            'business_insight': 0.20,
            'leadership': 0.20,
            'technical_literacy': 0.15,
            'communication': 0.10,
            'cultural_fit': 0.10
        }

    def calculate_total_score(self, scores: Dict[str, int]) -> float:
        """
        计算加权总分
        scores: {'strategic_thinking': 4, 'business_insight': 5, ...}
        """
        total_score = 0.0
        for dimension, weight in self.dimensions.items():
            score = scores.get(dimension, 0)
            total_score += score * weight

        return round(total_score, 2)

    def generate_evaluation_report(self, candidate_id: str, scores: Dict[str, int]) -> str:
        """生成评估报告"""
        total_score = self.calculate_total_score(scores)

        report = f"""
        候选人评估报告 - {candidate_id}

        综合得分: {total_score}/5.0

        维度得分详情:
        """

        for dimension, weight in self.dimensions.items():
            score = scores.get(dimension, 0)
            weighted_score = score * weight
            report += f"- {dimension}: {score}/5 (权重{weight:.0%}) = {weighted_score:.2f}\n"

        # 生成录用建议
        if total_score >= 4.0:
            recommendation = "强烈推荐录用"
        elif total_score >= 3.5:
            recommendation = "推荐录用"
        elif total_score >= 3.0:
            recommendation = "条件录用，需进一步评估"
        else:
            recommendation = "暂不推荐录用"

        report += f"\n录用建议: {recommendation}"

        return report
```
