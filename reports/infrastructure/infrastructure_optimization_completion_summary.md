```

│   ├── health/             # 健康检查

│   ├── resource_management/ # 资源管理
│   ├── performance/        # 性能测试
│   └── utils/              # 核心工具
├── services/               # 服务层
│   ├── database/           # 数据库服务
│   ├── cache/              # 缓存服务
│   ├── security/           # 安全服务
│   └── deployment/         # 部署服务
├── utils/                  # 工具层
│   ├── interfaces/         # 接口定义
│   ├── exceptions/         # 异常定义
│   ├── helpers/            # 辅助工具
│   └── validators/         # 验证工具
```
