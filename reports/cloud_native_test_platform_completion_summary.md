#### YAML生成
```python
def _create_deployment_yaml(self, name: str, config: KubernetesConfig, 
                           container_config: ContainerConfig) -> Dict[str, Any]:
    return {
        "apiVersion": "apps/v1",
        "kind": "Deployment",
        "metadata": {"name": name, "namespace": config.namespace},
        "spec": {
            "replicas": config.replicas,
            "selector": {"matchLabels": {"app": name}},
            "template": {
                "metadata": {"labels": {"app": name}},
                "spec": {"containers": [{
                    "name": name,
                    "image": f"{container_config.image}:{container_config.tag}",
                    "ports": [{"containerPort": int(port)} for port in container_config.ports.values()],
                    "resources": {
                        "limits": config.resource_limits,
                        "requests": config.resource_requests
                    }
                }]}
            }
        }
    }
```
