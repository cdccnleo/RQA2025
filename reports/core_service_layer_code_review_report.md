```python
# 重构前 (复杂度30)
def shutdown(self):
    if self._running:
        try:
            self._cleanup_resources()
            if self._workers:
                for worker in self._workers:
                    worker.stop()
            self._save_state()
        except Exception as e:
            self._handle_shutdown_error(e)
    self._running = False

# 重构后 (复杂度<10)
def shutdown(self):
    """优雅关闭服务"""
    if not self._running:
        return

    self._running = False
    self._perform_graceful_shutdown()
    self._save_final_state()

def _perform_graceful_shutdown(self):
    """执行优雅关闭逻辑"""
    self._cleanup_resources()
    self._stop_all_workers()
    self._handle_shutdown_errors()
```
