### A. 监控脚本示例

```python
# check_file_size.py
import os
from pathlib import Path

def check_trading_layer_files():
    """检查交易层文件大小"""
    
    trading_root = Path("src/trading")
    warnings = []
    errors = []
    
    for py_file in trading_root.rglob("*.py"):
        if "__pycache__" in str(py_file):
            continue
        
        with open(py_file, 'r', encoding='utf-8') as f:
            lines = len(f.readlines())
        
        # 检查阈值
        if lines > 1000:
            errors.append(f"{py_file}: {lines}行 (超过红线)")
        elif lines > 800:
            warnings.append(f"{py_file}: {lines}行 (接近上限)")
    
    # 输出结果
    if errors:
        print("🔴 红色预警:")
        for e in errors:
            print(f"  {e}")
    
    if warnings:
        print("🟡 黄色预警:")
        for w in warnings:
            print(f"  {w}")
    
    if not errors and not warnings:
        print("✅ 所有文件大小正常")
    
    return len(errors) == 0

if __name__ == "__main__":
    check_trading_layer_files()
```
