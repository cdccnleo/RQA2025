
#### 2.2 快速上手

**示例1：创建一个简单组件**

```python
from src.core.foundation.base_component import BaseComponent, component

@component("data_processor")
class DataProcessorComponent(BaseComponent):
    """数据处理组件"""
    
    def _do_initialize(self, config):
        """初始化：只需实现特定逻辑"""
        self.batch_size = config.get('batch_size', 100)
        self.processors = []
        return True  # 返回True表示成功
    
    def _do_execute(self, *args, **kwargs):
        """执行：只需实现业务逻辑"""
        data = kwargs.get('data', [])
        operation = kwargs.get('operation', 'process')
        
        if operation == 'process':
            return self._process_data(data)
        elif operation == 'batch':
            return self._batch_process(data)
        
        return None
    
    def _process_data(self, data):
        """具体业务逻辑"""
        return [item * 2 for item in data]
    
    def _batch_process(self, data):
        """批处理逻辑"""
        batches = [data[i:i+self.batch_size] 
                  for i in range(0, len(data), self.batch_size)]
        return [self._process_data(batch) for batch in batches]

# 使用示例
component = DataProcessorComponent("processor")
component.initialize({'batch_size': 50})

# 执行操作
result = component.execute(operation='process', data=[1, 2, 3, 4, 5])
print(result)  # [2, 4, 6, 8, 10]

# 获取状态
info = component.get_info()
print(info['status'])  # initialized

# 获取统计
stats = component.get_info()
print(stats)
```

**关键点**:
- ✅ 只需实现`_do_initialize`和`_do_execute`
- ✅ 日志、状态、错误处理自动处理
- ✅ 使用装饰器更简洁

#### 2.3 实战练习

**练习1**: 创建一个缓存组件
```python
@component("cache")
class CacheComponent(BaseComponent):
    def _do_initialize(self, config):
        self._cache = {}
        self.max_size = config.get('max_size', 1000)
        return True
    
    def _do_execute(self, *args, **kwargs):
        op = kwargs.get('operation')
        key = kwargs.get('key')
        
        if op == 'get':
            return self._cache.get(key)
        elif op == 'set':
            value = kwargs.get('value')
            if len(self._cache) < self.max_size:
                self._cache[key] = value
            return True
        elif op == 'clear':
            self._cache.clear()
            return True
```

#### 3.2 快速上手

**示例2：创建一个数据适配器**

```python
from src.core.foundation.base_adapter import BaseAdapter, adapter
from typing import Dict
from decimal import Decimal

@adapter("price_adapter", enable_cache=True)
class PriceAdapter(BaseAdapter[Dict, Dict]):
    """价格数据适配器"""
    
    def _do_adapt(self, data: Dict) -> Dict:
        """核心适配逻辑"""
        return {
            'symbol': data['code'].upper(),
            'price': Decimal(str(data['price'])),
            'volume': int(data['vol']),
            'timestamp': data['time']
        }
    
    def validate_input(self, data: Dict) -> bool:
        """输入验证"""
        required = ['code', 'price', 'vol', 'time']
        return all(field in data for field in required)
    
    def _preprocess(self, data: Dict) -> Dict:
        """预处理：清理数据"""
        cleaned = data.copy()
        cleaned['code'] = str(data['code']).strip()
        return cleaned
    
    def _postprocess(self, data: Dict) -> Dict:
        """后处理：添加元数据"""
        data['adapted_at'] = datetime.now()
        return data

# 使用示例
adapter = PriceAdapter(name="price")

# 适配数据
input_data = {'code': 'sh600000', 'price': '10.5', 'vol': '1000', 'time': '2025-11-03'}
output_data = adapter.adapt(input_data)

print(output_data)
# {
#     'symbol': 'SH600000',
#     'price': Decimal('10.5'),
#     'volume': 1000,
#     'timestamp': '2025-11-03',
#     'adapted_at': datetime(...)
# }

# 查看统计
stats = adapter.get_stats()
print(f"成功率: {stats['success_rate']}")
print(f"缓存大小: {stats['cache_size']}")
```

**关键点**:
- ✅ 只需实现`_do_adapt`（必须）
- ✅ 可选实现`validate_input`、`_preprocess`、`_postprocess`
- ✅ 缓存、统计、错误处理自动处理

```python
from src.core.foundation.base_adapter import AdapterChain

# 创建处理链
class Validator(BaseAdapter[Dict, Dict]):

            raise ValueError("缺少必需字段")

chain.add_adapter(Enricher("enricher"))

# 执行

# result = {'required_field': 'value', 'other': 'data', 'enriched_at': ..., 'version': '2.0'}
```

#### 4.2 错误恢复

**场景**：适配失败时的降级处理

```python
class RobustAdapter(BaseAdapter[Dict, Dict]):
    def _do_adapt(self, data: Dict) -> Dict:
        if data.get('force_error'):
            raise ValueError("强制错误")
        return {'processed': True}
    
    def _handle_error(self, data: Dict, error: Exception) -> Dict:
        """自定义错误恢复"""
        self._logger.warning(f"适配失败，使用默认值: {error}")
        return {
            'processed': False,
            'error': str(error),
            'fallback': True
        }

# 使用
adapter = RobustAdapter("robust")
result = adapter.adapt({'force_error': True})
# result = {'processed': False, 'error': '强制错误', 'fallback': True}
```

**业务场景**：处理交易数据

**实现**:
```python
from src.core.foundation.base_adapter import AdapterChain

# 定义各步骤适配器
@adapter("validator")
class DataValidator(BaseAdapter[Dict, Dict]):
    def _do_adapt(self, data: Dict) -> Dict:
        required_fields = ['symbol', 'price', 'quantity']
        if not all(f in data for f in required_fields):
            raise ValueError("数据不完整")
        return data

@adapter("normalizer")
class DataNormalizer(BaseAdapter[Dict, Dict]):
    def _do_adapt(self, data: Dict) -> Dict:
        return {
            'symbol': data['symbol'].upper(),
            'price': Decimal(str(data['price'])),
            'quantity': int(data['quantity'])
        }

@adapter("enricher")
class DataEnricher(BaseAdapter[Dict, Dict]):
    def _do_adapt(self, data: Dict) -> Dict:
        data['timestamp'] = datetime.now()
        data['source'] = 'trading_system'
        return data

@adapter("encryptor")
class DataEncryptor(BaseAdapter[Dict, Dict]):
    def _do_adapt(self, data: Dict) -> Dict:
        # 加密敏感字段
        if 'account_id' in data:
            data['account_id'] = self._encrypt(data['account_id'])
        return data

# 构建处理链
pipeline = AdapterChain("trading_data_pipeline")
pipeline.add_adapter(DataValidator("val"))
pipeline.add_adapter(DataNormalizer("norm"))
pipeline.add_adapter(DataEnricher("enrich"))
pipeline.add_adapter(DataEncryptor("encrypt"))

# 执行

processed_data = pipeline.execute(raw_data)

```
