### API集成

```javascript
// API基础URL生成（环境感知）
function getApiBaseUrl(endpoint = '') {
    const baseUrl = window.location.protocol === 'file:' || window.location.hostname === 'localhost'
        ? 'http://localhost:8000/api/v1'
        : '/api/v1';
    return baseUrl + endpoint;
}
```
