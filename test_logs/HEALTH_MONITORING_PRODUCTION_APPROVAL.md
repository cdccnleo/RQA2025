## 🚀 投产部署配置

### 环境配置

```python
# 生产环境启用配置
HEALTH_MONITORING_CONFIG = {
    # 核心监控模块（全部启用）
    'core_modules': {
        'connection_health_checker': {
            'enabled': True,
            'check_interval': 60,
        },
        'connection_pool_monitor': {
            'enabled': True,
            'report_interval': 300,
        },
        'storage_monitor_plugin': {
            'enabled': True,
            'check_interval': 120,
        },
    },
    
    # 基础日志（基础功能启用）
    'logger': {
        'enabled': True,
        'mode': 'basic',
        'level': 'INFO',
    },
    
    # 日志插件（全部禁用，使用降级方案）
    'log_plugins': {
        'market_data_logger': {'enabled': False},
        'log_compressor': {'enabled': False},
        'log_backpressure': {'enabled': False},
    },
}
```

### 使用示例

```python
# 示例1：使用核心健康检查
from src.infrastructure.utils.components.connection_health_checker import ConnectionHealthChecker

checker = ConnectionHealthChecker()
health_status = checker.health_check(connection_pool)
print(f"健康状态: {health_status}")

# 示例2：使用连接池监控
from src.infrastructure.utils.components.connection_pool_monitor import ConnectionPoolMonitor

monitor = ConnectionPoolMonitor()
monitor.record_acquire(connection_id="conn_001")
stats = monitor.get_statistics()
print(f"连接池统计: {stats}")

# 示例3：使用基础日志
from src.infrastructure.utils.monitoring.logger import get_logger

logger = get_logger("my_app")
logger.info("应用启动成功")
logger.error("发生错误", exc_info=True)
```
