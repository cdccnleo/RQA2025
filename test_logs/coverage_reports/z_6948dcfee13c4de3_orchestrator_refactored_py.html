113 def start_trading_cycle(self, symbols: List[str], strategy_config: dict, 

114 process_id: str = None) -> str: 

115 """ 

116 启动交易周期 

117  

118 Args: 

119 symbols: 交易标的列表 

120 strategy_config: 策略配置 

121 process_id: 流程ID（可选） 

122  

123 Returns: 

124 str: 实例ID 

125 """ 

126 # 创建流程配置 

127 if process_id is None: 

128 process_id = f"process_{int(datetime.now().timestamp() * 1000)}" 

129 

130 config = create_process_config( 

131 process_id=process_id, 

132 name=f"Trading-{'-'.join(symbols[:3])}", 

133 parameters={'symbols': symbols, 'strategy': strategy_config} 

134 ) 

135 

136 # 从池获取实例 

137 instance = self.pool.get_instance(config) 

138 

139 # 注册到监控器 

140 self.monitor.register_process(instance) 

141 

142 # 发布启动事件 

143 self.event_bus.publish(EventType.PROCESS_STARTED, { 

144 'instance_id': instance.instance_id, 

145 'process_id': process_id, 

146 'symbols': symbols 

147 }) 

148 

149 # 状态转换 

150 self.state_machine.transition_to(BusinessProcessState.DATA_COLLECTING) 

151 

152 logger.info(f"交易周期已启动: {instance.instance_id}") 

153 return instance.instance_id

155 def complete_process(self, instance_id: str, 

156 final_status: BusinessProcessState = BusinessProcessState.COMPLETED) -> bool: 

157 """完成流程""" 

158 instance = self.monitor.get_process(instance_id) 

159 if not instance: 

160 logger.warning(f"流程不存在: {instance_id}") 

161 return False 

162 

163 # 更新状态 

164 instance.update_status(final_status) 

165 self.monitor.update_process(instance_id, final_status) 

166 

167 # 发布完成事件 

168 self.event_bus.publish(EventType.PROCESS_COMPLETED, { 

169 'instance_id': instance_id, 

170 'status': final_status.value 

171 }) 

172 

173 # 归还实例到池 

174 self.pool.return_instance(instance) 

175 

176 logger.info(f"流程已完成: {instance_id}") 

177 return True
