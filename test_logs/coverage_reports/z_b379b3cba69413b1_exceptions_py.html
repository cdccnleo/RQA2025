90def handle_resilience_exception(func): 

91 """ 

92 装饰器：统一处理弹性异常 

93 

94 Args: 

95 func: 被装饰的函数 

96 

97 Returns: 

98 包装后的函数 

99 """ 

100 def wrapper(*args, **kwargs): 

101 try: 

102 return func(*args, **kwargs) 

103 except ResilienceException: 

104 # 重新抛出弹性异常 

105 raise 

106 except Exception as e: 

107 # 将其他异常包装为弹性异常 

108 raise ResilienceException(f"意外弹性错误: {str(e)}") from e 

109 

110 return wrapper
