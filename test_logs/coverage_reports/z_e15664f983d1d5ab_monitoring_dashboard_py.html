284 """获取面板数据""" 

285 with self._lock: 

286 now = time.time() 

287 

288 # 计算统计信息 

289 stats = {}

291 if metrics: 

292 values = [m.value for m in metrics] 

293 stats[name] = { 

294 'count': len(values), 

295 'min': min(values),

297 'avg': statistics.mean(values), 

298 'latest': metrics[-1].value if metrics else 0

303 # 获取未解决的告警列表（在同一个锁保护下） 

304 unresolved_alerts = [a for a in self._alerts.values() if not a.resolved] 

305 recent_unresolved_alerts = sorted(unresolved_alerts, key=lambda x: x.timestamp, reverse=True)[:RECENT_ALERTS_LIMIT] 

306 

307 for alert in self._alerts.values(): 

308 alert_stats[alert.severity.value] += 1 

309 

310 return { 

311 'timestamp': now, 

312 'metrics_count': sum(len(metrics) for metrics in self._metrics.values()), 

313 'alerts_count': len(unresolved_alerts), 

314 'alert_rules_count': len(self._alert_rules), 

315 'stats': stats, 

316 'alert_stats': dict(alert_stats),

702 def import_data(self, file_path: str): 

703 """导入监控数据""" 

704 if not os.path.exists(file_path): 

705 logger.error(f"文件不存在: {file_path}") 

706 return 

707 

708 with open(file_path, 'r', encoding='utf - 8') as f: 

709 data = json.load(f) 

710 

711 with self._lock: 

712 # 导入指标 

713 for name, metrics_data in data.get('metrics', {}).items(): 

714 self._metrics[name] = deque(metrics_data, maxlen=self.config.max_metrics) 

715 

716 # 导入告警 

717 for alert_data in data.get('alerts', []): 

718 alert = Alert(**alert_data) 

719 self._alerts[alert.name] = alert 

720 

721 # 导入告警规则 

722 self._alert_rules.update(data.get('alert_rules', {})) 

723 

724 logger.info(f"监控数据已从 {file_path} 导入")

726 def _monitor_loop(self): 

727 """监控循环""" 

728 while self._running: 

729 try: 

730 self._check_alert_rules() 

731 # 使用更短的睡眠时间，以便更快响应停止信号 

732 for _ in range(int(self.config.refresh_interval * REFRESH_CHECK_MULTIPLIER)): # 每0.1秒检查一次 

733 if not self._running: 

734 break 

735 time.sleep(0.1) 

736 except Exception as e: 

737 logger.error(f"监控循环异常: {e}") 

738 if not self._running: 

739 break 

740 time.sleep(1.0) 

741 

742 def _cleanup_loop(self): 

743 """清理循环""" 

744 while self._running: 

745 try: 

746 self._cleanup_old_data() 

747 # 使用更短的睡眠时间，以便更快响应停止信号 

748 for _ in range(STOP_CHECK_INTERVAL): # 每分钟检查一次，但每0.1秒检查停止信号 

749 if not self._running: 

750 break 

751 time.sleep(0.1) 

752 except Exception as e: 

753 logger.error(f"清理循环异常: {e}") 

754 if not self._running: 

755 break 

756 time.sleep(STOP_CHECK_DELAY)

758 def _check_alert_rules(self): 

759 """检查告警规则""" 

760 with self._lock: 

761 for rule_name, rule in self._alert_rules.items(): 

762 try: 

763 # 简单的规则检查逻辑 

764 if rule['query'] in self._metrics: 

765 metrics = self._metrics[rule['query']] 

766 if metrics: 

767 latest_value = metrics[-1].value 

768 

769 # 检查阈值 

770 if latest_value > rule['threshold']: 

771 alert_name = f"{rule_name}_{int(time.time())}" 

772 alert = Alert( 

773 name=alert_name, 

774 message=f"指标 {rule['query']} 值 {latest_value} 超过阈值 {rule['threshold']}", 

775 severity=rule['severity'], 

776 source=rule['query'] 

777 ) 

778 

779 self._alerts[alert_name] = alert 

780 

781 # 触发告警回调 

782 for callback in self._alert_callbacks: 

783 try: 

784 callback(alert) 

785 except Exception as e: 

786 logger.error(f"告警回调执行失败: {e}") 

787 

788 except Exception as e: 

789 logger.error(f"检查告警规则 {rule_name} 失败: {e}") 

790 

791 def _cleanup_old_data(self): 

792 """清理旧数据""" 

793 if not self.config.enable_auto_cleanup: 

794 return 

795 

796 cutoff_time = time.time() - (self.config.retention_days * HOURS_PER_DAY * SECONDS_PER_HOUR) 

797 

798 with self._lock: 

799 # 清理旧指标 

800 for name in list(self._metrics.keys()): 

801 metrics = self._metrics[name] 

802 while metrics and metrics[0].timestamp < cutoff_time: 

803 metrics.popleft() 

804 

805 # 如果指标为空，删除整个键 

806 if not metrics: 

807 del self._metrics[name] 

808 

809 # 清理已解决的旧告警 

810 current_time = time.time() 

811 for alert_name in list(self._alerts.keys()): 

812 alert = self._alerts[alert_name] 

813 if alert.resolved and (current_time - alert.timestamp) > self.config.alert_timeout: 

814 del self._alerts[alert_name]
