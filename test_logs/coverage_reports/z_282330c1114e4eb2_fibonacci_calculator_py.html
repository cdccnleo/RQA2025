14class FibonacciCalculator: 

15 

16 """斐波那契计算器""" 

17 

18 # 标准斐波那契水平 

19 FIB_LEVELS = [0.0, 0.236, 0.382, 0.500, 0.618, 0.786, 1.0] 

20 FIB_EXTENSIONS = [0.0, 0.236, 0.382, 0.500, 0.618, 0.786, 1.0, 1.272, 1.414, 1.618, 2.0, 2.618] 

21 

22 def __init__(self, config: Optional[Dict[str, Any]] = None): 

23 

24 self.config = config or {} 

25 self.lookback_period = self.config.get('lookback_period', 50) 

26 self.min_swing_length = self.config.get('min_swing_length', 5) 

27 self.custom_levels = self.config.get('custom_levels', None) 

28 

29 def calculate(self, data: pd.DataFrame) -> pd.DataFrame: 

30 """ 

31 计算斐波那契水平 

32 

33 Args: 

34 data: 包含OHLC数据的DataFrame 

35 

36 Returns: 

37 包含斐波那契水平的DataFrame 

38 """ 

39 try: 

40 if data is None or data.empty: 

41 logger.warning("输入数据为空") 

42 return pd.DataFrame() 

43 

44 result_df = data.copy() 

45 

46 # 确保必要列存在 

47 required_columns = ['high', 'low', 'close'] 

48 if not all(col in result_df.columns for col in required_columns): 

49 logger.error( 

50 f"数据缺少必要列: {[col for col in required_columns if col not in result_df.columns]}") 

51 return result_df 

52 

53 # 计算摆动点 

54 swing_highs, swing_lows = self._find_swing_points(result_df) 

55 

56 # 计算回撤水平 

57 fib_levels = self._calculate_fibonacci_levels(result_df, swing_highs, swing_lows) 

58 

59 # 添加到结果DataFrame 

60 for level_name, level_values in fib_levels.items(): 

61 result_df[level_name] = level_values 

62 

63 # 计算价格与斐波那契水平的关系 

64 result_df = self._calculate_price_fib_relationship(result_df, fib_levels) 

65 

66 logger.info("斐波那契水平计算完成") 

67 return result_df 

68 

69 except Exception as e: 

70 logger.error(f"斐波那契计算失败: {e}") 

71 return data

73 def _find_swing_points(self, data: pd.DataFrame) -> tuple: 

74 """寻找摆动点""" 

75 highs = data['high'] 

76 lows = data['low'] 

77 

78 swing_highs = [] 

79 swing_lows = [] 

80 

81 for i in range(self.min_swing_length, len(data) - self.min_swing_length): 

82 # 检查高点 

83 if highs.iloc[i] == highs.iloc[i - self.min_swing_length:i + self.min_swing_length + 1].max(): 

84 swing_highs.append((i, highs.iloc[i])) 

85 

86 # 检查低点 

87 if lows.iloc[i] == lows.iloc[i - self.min_swing_length:i + self.min_swing_length + 1].min(): 

88 swing_lows.append((i, lows.iloc[i])) 

89 

90 return swing_highs, swing_lows

92 def _calculate_fibonacci_levels(self, data: pd.DataFrame, 

93 

94 

95 swing_highs: List[tuple], 

96 swing_lows: List[tuple]) -> Dict[str, pd.Series]: 

97 """计算斐波那契水平""" 

98 fib_levels = {} 

99 

100 if not swing_highs or not swing_lows: 

101 # 如果没有找到摆动点，使用最近的高低点 

102 recent_high = data['high'].rolling(window=self.lookback_period).max().iloc[-1] 

103 recent_low = data['low'].rolling(window=self.lookback_period).min().iloc[-1] 

104 swing_highs = [(len(data) - 1, recent_high)] 

105 swing_lows = [(len(data) - 1, recent_low)] 

106 

107 # 找到最近的主要摆动点 

108 recent_swing_high = max(swing_highs, key=lambda x: x[0]) 

109 recent_swing_low = min(swing_lows, key=lambda x: x[0]) 

110 

111 high_price = recent_swing_high[1] 

112 low_price = recent_swing_low[1] 

113 

114 # 计算回撤水平 

115 levels_to_use = self.custom_levels or self.FIB_LEVELS 

116 

117 for level in levels_to_use: 

118 level_name = f'fib_retrace_{level:.3f}' 

119 fib_levels[level_name] = low_price + (high_price - low_price) * level 

120 

121 # 计算扩展水平（如果有足够的数据） 

122 if len(swing_highs) >= 2 and len(swing_lows) >= 2: 

123 # 找到前一个主要摆动 

124 prev_swing_high = max([h for h in swing_highs if h[0] < recent_swing_high[0]], 

125 key=lambda x: x[0], default=None) 

126 prev_swing_low = min([l for l in swing_lows if l[0] < recent_swing_low[0]], 

127 key=lambda x: x[0], default=None) 

128 

129 if prev_swing_high and prev_swing_low: 

130 extension_range = abs(high_price - low_price) 

131 

132 for ext_level in self.FIB_EXTENSIONS: 

133 level_name = f'fib_ext_{ext_level:.3f}' 

134 if recent_swing_high[0] > recent_swing_low[0]: # 上涨趋势 

135 fib_levels[level_name] = low_price + extension_range * ext_level 

136 else: # 下跌趋势 

137 fib_levels[level_name] = high_price - extension_range * ext_level 

138 

139 # 将水平转换为Series 

140 for level_name in fib_levels: 

141 fib_levels[level_name] = pd.Series([fib_levels[level_name]] * len(data), 

142 index=data.index) 

143 

144 return fib_levels

146 def _calculate_price_fib_relationship(self, data: pd.DataFrame, 

147 

148 

149 fib_levels: Dict[str, pd.Series]) -> pd.DataFrame: 

150 """计算价格与斐波那契水平的关系""" 

151 close_price = data['close'] 

152 

153 # 计算距离各个水平的距离 

154 for level_name, level_values in fib_levels.items(): 

155 distance_col = f'{level_name}_distance' 

156 data[distance_col] = close_price - level_values 

157 

158 # 计算是否接近水平（在一定范围内） 

159 tolerance = close_price * 0.005 # 0.5 % 的容差 

160 near_level_col = f'{level_name}_near' 

161 data[near_level_col] = abs(data[distance_col]) <= tolerance 

162 

163 # 计算最接近的水平 

164 data['fib_nearest_level'] = None 

165 data['fib_nearest_distance'] = float('inf') 

166 

167 for level_name, level_values in fib_levels.items(): 

168 distance = abs(close_price - level_values) 

169 closer_mask = distance < data['fib_nearest_distance'] 

170 

171 data.loc[closer_mask, 'fib_nearest_level'] = level_name.replace( 

172 'fib_', '').replace('_', ' ') 

173 data.loc[closer_mask, 'fib_nearest_distance'] = distance[closer_mask] 

174 

175 # 计算支撑和阻力水平 

176 data['fib_support_level'] = None 

177 data['fib_resistance_level'] = None 

178 

179 # 简单支撑阻力识别（可以根据需要扩展） 

180 for i in range(len(data)): 

181 current_price = close_price.iloc[i] 

182 

183 # 寻找低于当前价格的回撤水平作为支撑 

184 support_levels = [] 

185 resistance_levels = [] 

186 

187 for level_name, level_values in fib_levels.items(): 

188 level_price = level_values.iloc[i] 

189 if level_price < current_price and 'retrace' in level_name: 

190 support_levels.append((level_name, level_price)) 

191 elif level_price > current_price and 'retrace' in level_name: 

192 resistance_levels.append((level_name, level_price)) 

193 

194 # 选择最近的支撑和阻力 

195 if support_levels: 

196 nearest_support = max(support_levels, key=lambda x: x[1]) 

197 data.at[data.index[i], 'fib_support_level'] = nearest_support[0] 

198 

199 if resistance_levels: 

200 nearest_resistance = min(resistance_levels, key=lambda x: x[1]) 

201 data.at[data.index[i], 'fib_resistance_level'] = nearest_resistance[0] 

202 

203 return data
