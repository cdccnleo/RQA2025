53def standardize_metrics_format(raw_metrics: Dict[str, Any]) -> Dict[str, Any]: 

54 """ 

55 标准化指标数据格式 

56 

57 将各种来源的指标数据转换为统一的格式标准。 

58 

59 Args: 

60 raw_metrics: 原始指标数据字典 

61 

62 Returns: 

63 Dict[str, Any]: 标准化的指标数据 

64 """ 

65 try: 

66 # 确保基本字段存在 

67 standardized = { 

68 "timestamp": raw_metrics.get("timestamp", datetime.now().isoformat()), 

69 "source": raw_metrics.get("source", SOURCE_SYSTEM), 

70 "version": "1.0", # 数据格式版本 

71 "metadata": raw_metrics.get("metadata", {}), 

72 "metrics": {} 

73 } 

74 

75 # 标准化各个指标类型 

76 raw_data = raw_metrics.get("metrics", raw_metrics) 

77 

78 # CPU指标标准化 

79 if "cpu" in raw_data: 

80 standardized["metrics"]["cpu"] = _standardize_cpu_metrics(raw_data["cpu"]) 

81 

82 # 内存指标标准化 

83 if "memory" in raw_data: 

84 standardized["metrics"]["memory"] = _standardize_memory_metrics(raw_data["memory"]) 

85 

86 # 磁盘指标标准化 

87 if "disk" in raw_data: 

88 standardized["metrics"]["disk"] = _standardize_disk_metrics(raw_data["disk"]) 

89 

90 # 网络指标标准化 

91 if "network" in raw_data: 

92 standardized["metrics"]["network"] = _standardize_network_metrics(raw_data["network"]) 

93 

94 # GPU指标标准化 

95 if "gpu" in raw_data: 

96 standardized["metrics"]["gpu"] = _standardize_gpu_metrics(raw_data["gpu"]) 

97 

98 # 系统指标标准化 

99 if "system" in raw_data: 

100 standardized["metrics"]["system"] = _standardize_system_metrics(raw_data["system"]) 

101 

102 return standardized 

103 

104 except Exception as e: 

105 logger.error(f"指标数据格式标准化失败: {e}") 

106 return { 

107 "timestamp": datetime.now().isoformat(), 

108 "source": SOURCE_SYSTEM, 

109 "version": "1.0", 

110 "error": str(e), 

111 "original_data": raw_metrics

367 """

379 {

381 "function": "出错的函数名",

383 } 

384 

385 使用示例： 

386 @handle_metrics_exceptions 

387 def collect_cpu_metrics(self): 

388 # 收集逻辑 

389 return metrics_data 

390 

391 # 如果发生异常，会自动返回标准化的错误格式 

392 # 而不是抛出异常 

393 """ 

394 def wrapper(*args, **kwargs): 

395 try: 

396 return func(*args, **kwargs) 

397 except PermissionError:

404 except ConnectionError:

407 "error": "connection_failed", 

408 "function": func.__name__, 

409 "timestamp": datetime.now().isoformat() 

410 } 

411 except TimeoutError: 

412 logger.warning(f"操作超时: {func.__name__}") 

413 return { 

414 "error": "timeout", 

415 "function": func.__name__, 

416 "timestamp": datetime.now().isoformat() 

417 } 

418 except Exception as e: 

419 logger.error(f"指标收集异常 {func.__name__}: {e}")

421 "error": str(e), 

422 "function": func.__name__,

428def validate_metrics_data(metrics: Dict[str, Any]) -> Dict[str, Any]: 

429 """ 

430 验证指标数据完整性和质量 

431 

432 对指标数据进行全面验证，包括： 

433 - 必要字段检查（timestamp, source, metrics） 

434 - 时间戳格式验证 

435 - 数据类型验证 

436 - 数值范围检查 

437 - 质量评分计算 

438 

439 Args: 

440 metrics: 要验证的指标数据，必须是标准化的格式 

441 

442 Returns: 

443 Dict[str, Any]: 验证结果，包含： 

444 - valid: bool，数据是否有效 

445 - errors: List[str]，错误列表 

446 - warnings: List[str]，警告列表 

447 - quality_score: float，质量评分(0.0-1.0) 

448 

449 示例： 

450 metrics = { 

451 "timestamp": "2025-09-28T10:00:00", 

452 "source": "system", 

453 "metrics": {"cpu": {"usage_percent": 85.5}} 

454 } 

455 

456 result = validate_metrics_data(metrics) 

457 # result = { 

458 # "valid": True, 

459 # "errors": [], 

460 # "warnings": [], 

461 # "quality_score": 1.0 

462 # } 

463 """ 

464 errors = [] 

465 warnings = [] 

466 

467 # 检查必要字段 

468 required_fields = ["timestamp", "source", "metrics"] 

469 for field in required_fields: 

470 if field not in metrics: 

471 errors.append(f"缺少必要字段: {field}") 

472 

473 # 检查时间戳格式 

474 if "timestamp" in metrics: 

475 try: 

476 datetime.fromisoformat(metrics["timestamp"].replace('Z', '+00:00')) 

477 except (ValueError, TypeError): 

478 errors.append("时间戳格式无效") 

479 

480 # 检查指标数据 

481 if "metrics" in metrics and isinstance(metrics["metrics"], dict): 

482 metrics_data = metrics["metrics"] 

483 

484 # 检查各指标类型的合理性 

485 for metric_type, metric_data in metrics_data.items(): 

486 if metric_type == "cpu": 

487 _validate_cpu_metrics(metric_data, errors, warnings) 

488 elif metric_type == "memory": 

489 _validate_memory_metrics(metric_data, errors, warnings) 

490 elif metric_type == "disk": 

491 _validate_disk_metrics(metric_data, errors, warnings) 

492 elif metric_type == "network": 

493 _validate_network_metrics(metric_data, errors, warnings) 

494 else: 

495 errors.append("metrics字段缺失或格式错误") 

496 

497 return { 

498 "valid": len(errors) == 0, 

499 "errors": errors, 

500 "warnings": warnings, 

501 "quality_score": max(0.0, 1.0 - (len(errors) * 0.2 + len(warnings) * 0.1)) 

502 }
