92class MarketRegimeDetector: 

93 

94 """市场状态检测器"""

97 

98 self.regime_history = [] 

99 self.current_regime = MarketRegime.NORMAL 

100 self.regime_confidence = 0.0 

101 

102 # 市场状态阈值 

103 self.thresholds = { 

104 'volatility_normal': 0.15, 

105 'volatility_volatile': 0.25, 

106 'volatility_crisis': 0.40, 

107 'stress_normal': 0.3, 

108 'stress_crisis': 0.7, 

109 'correlation_normal': 0.5, 

110 'correlation_crisis': 0.8

113 def detect_regime(self, market_data: Dict[str, Any]) -> Tuple[MarketRegime, float]: 

114 """检测市场状态""" 

115 # 提取关键指标 

116 volatility = market_data.get('market_volatility', 0.15) 

117 stress_level = market_data.get('market_stress_level', 0.3) 

118 correlation = market_data.get('average_correlation', 0.3) 

119 price_momentum = market_data.get('price_momentum', 0.0) 

120 volume_spike = market_data.get('volume_spike', False) 

121 

122 # 计算市场状态得分 

123 volatility_score = self._calculate_volatility_score(volatility) 

124 stress_score = self._calculate_stress_score(stress_level) 

125 correlation_score = self._calculate_correlation_score(correlation) 

126 

127 # 综合评分 

128 total_score = (volatility_score + stress_score + correlation_score) / 3 

129 

130 # 确定市场状态 

131 if total_score >= 0.8 or volume_spike: 

132 regime = MarketRegime.CRISIS 

133 confidence = min(total_score, 0.95) 

134 elif total_score >= 0.6: 

135 regime = MarketRegime.STRESSED 

136 confidence = total_score 

137 elif total_score >= 0.4: 

138 regime = MarketRegime.VOLATILE 

139 confidence = total_score 

140 elif price_momentum > 0.1: 

141 regime = MarketRegime.RECOVERY 

142 confidence = 0.7 

143 else: 

144 regime = MarketRegime.NORMAL 

145 confidence = 1.0 - total_score 

146 

147 self.current_regime = regime 

148 self.regime_confidence = confidence 

149 

150 # 记录历史 

151 self.regime_history.append({ 

152 'timestamp': datetime.now().isoformat(), 

153 'regime': regime.value, 

154 'confidence': confidence, 

155 'scores': { 

156 'volatility': volatility_score, 

157 'stress': stress_score, 

158 'correlation': correlation_score 

159 } 

160 }) 

161 

162 # 限制历史记录数量 

163 if len(self.regime_history) > 1000: 

164 self.regime_history = self.regime_history[-1000:] 

165 

166 return regime, confidence

168 def _calculate_volatility_score(self, volatility: float) -> float: 

169 """计算波动率得分""" 

170 if volatility >= self.thresholds['volatility_crisis']: 

171 return 1.0 

172 elif volatility >= self.thresholds['volatility_volatile']: 

173 return 0.7 + (volatility - self.thresholds['volatility_volatile']) / \ 

174 (self.thresholds['volatility_crisis'] - 

175 self.thresholds['volatility_volatile']) * 0.3 

176 elif volatility >= self.thresholds['volatility_normal']: 

177 return 0.4 + (volatility - self.thresholds['volatility_normal']) / \ 

178 (self.thresholds['volatility_volatile'] - 

179 self.thresholds['volatility_normal']) * 0.3 

180 else: 

181 return volatility / self.thresholds['volatility_normal'] * 0.4 

182 

183 def _calculate_stress_score(self, stress_level: float) -> float: 

184 """计算压力得分""" 

185 if stress_level >= self.thresholds['stress_crisis']: 

186 return 1.0 

187 elif stress_level >= self.thresholds['stress_normal']: 

188 return 0.5 + (stress_level - self.thresholds['stress_normal']) / \ 

189 (self.thresholds['stress_crisis'] - self.thresholds['stress_normal']) * 0.5 

190 else: 

191 return stress_level / self.thresholds['stress_normal'] * 0.5 

192 

193 def _calculate_correlation_score(self, correlation: float) -> float: 

194 """计算相关性得分""" 

195 if correlation >= self.thresholds['correlation_crisis']: 

196 return 1.0 

197 elif correlation >= self.thresholds['correlation_normal']: 

198 return 0.6 + (correlation - self.thresholds['correlation_normal']) / \ 

199 (self.thresholds['correlation_crisis'] - 

200 self.thresholds['correlation_normal']) * 0.4 

201 else: 

202 return correlation / self.thresholds['correlation_normal'] * 0.6

204 def get_regime_statistics(self) -> Dict[str, Any]: 

205 """获取状态统计""" 

206 if not self.regime_history: 

207 return {} 

208 

209 recent_regimes = self.regime_history[-100:] # 最近100个记录 

210 

211 # 计算状态分布 

212 regime_counts = {} 

213 for record in recent_regimes: 

214 regime = record['regime'] 

215 regime_counts[regime] = regime_counts.get(regime, 0) + 1 

216 

217 # 计算状态持续时间 

218 current_regime_start = None 

219 regime_durations = [] 

220 

221 for i, record in enumerate(recent_regimes): 

222 if i == 0 or record['regime'] != recent_regimes[i - 1]['regime']: 

223 if current_regime_start is not None: 

224 duration = datetime.fromisoformat(record['timestamp']) - current_regime_start 

225 regime_durations.append({ 

226 'regime': recent_regimes[i - 1]['regime'], 

227 'duration_minutes': duration.total_seconds() / 60 

228 }) 

229 current_regime_start = datetime.fromisoformat(record['timestamp']) 

230 

231 return { 

232 'current_regime': self.current_regime.value, 

233 'regime_confidence': self.regime_confidence, 

234 'regime_distribution': regime_counts, 

235 'recent_transitions': regime_durations[-5:], # 最近5个状态转换 

236 'total_records': len(self.regime_history) 

237 }

278 def calculate_adjustment(self, asset_symbol: str, current_limits: Dict[str, float], 

279 

280 

281 market_data: Dict[str, Any]) -> LimitAdjustment: 

282 """计算限额调整""" 

283 # 检测市场状态 

284 market_regime, confidence = self.market_detector.detect_regime(market_data) 

285 

286 # 根据策略计算调整因子 

287 if self.strategy == LimitAdjustmentStrategy.ADAPTIVE: 

288 adjustment_factors = self._calculate_adaptive_factors(market_regime, market_data) 

289 else: 

290 adjustment_factors = self._calculate_strategy_factors(market_regime) 

291 

292 # 计算调整后的限额 

293 adjusted_limits = {} 

294 for limit_name, current_value in current_limits.items(): 

295 if limit_name in adjustment_factors: 

296 factor = adjustment_factors[limit_name] 

297 adjusted_limits[limit_name] = current_value * factor 

298 else: 

299 adjusted_limits[limit_name] = current_value 

300 

301 # 确定调整原因 

302 reason = self._generate_adjustment_reason(market_regime, confidence, market_data) 

303 

304 # 计算预期影响 

305 expected_impact = self._calculate_expected_impact( 

306 current_limits, adjusted_limits, market_data 

307 ) 

308 

309 # 创建调整对象 

310 adjustment = LimitAdjustment( 

311 adjustment_id=f"limit_adj_{asset_symbol}_{int(time.time())}", 

312 timestamp=datetime.now(), 

313 strategy=self.strategy, 

314 market_regime=market_regime, 

315 asset_symbol=asset_symbol, 

316 original_limits=current_limits.copy(), 

317 adjusted_limits=adjusted_limits, 

318 adjustment_factors=adjustment_factors, 

319 reason=reason, 

320 expected_impact=expected_impact 

321 ) 

322 

323 self.adjustments.append(adjustment) 

324 return adjustment

326 def _calculate_adaptive_factors(self, market_regime: MarketRegime, 

327 

328 

329 market_data: Dict[str, Any]) -> Dict[str, float]: 

330 """计算自适应调整因子""" 

331 base_factor = self.strategy_params[LimitAdjustmentStrategy.ADAPTIVE]['base_factor'] 

332 regime_sensitivity = self.strategy_params[LimitAdjustmentStrategy.ADAPTIVE]['regime_sensitivity'] 

333 

334 # 根据市场状态调整基础因子 

335 if market_regime == MarketRegime.CRISIS: 

336 regime_multiplier = 0.5 

337 elif market_regime == MarketRegime.STRESSED: 

338 regime_multiplier = 0.7 

339 elif market_regime == MarketRegime.VOLATILE: 

340 regime_multiplier = 0.85 

341 elif market_regime == MarketRegime.RECOVERY: 

342 regime_multiplier = 1.1 

343 else: # NORMAL 

344 regime_multiplier = 1.0 

345 

346 # 考虑动量因素 

347 momentum = market_data.get('price_momentum', 0.0) 

348 momentum_factor = 1.0 + momentum * 0.1 

349 

350 # 计算最终因子 

351 final_factor = base_factor * regime_multiplier * momentum_factor 

352 

353 return { 

354 'max_position_size': final_factor, 

355 'max_var_limit': final_factor, 

356 'max_drawdown_limit': final_factor * 0.9, # 回撤限额更保守 

357 'daily_loss_limit': final_factor * 0.8, # 每日损失限额最保守 

358 'volatility_limit': final_factor 

359 }

384 def _generate_adjustment_reason(self, market_regime: MarketRegime, confidence: float, 

385 

386 

387 market_data: Dict[str, Any]) -> str: 

388 """生成调整原因""" 

389 reasons = [] 

390 

391 if market_regime != MarketRegime.NORMAL: 

392 reasons.append(f"市场状态: {market_regime.value} (置信度: {confidence:.1%})") 

393 

394 volatility = market_data.get('market_volatility', 0) 

395 if volatility > 0.25: 

396 reasons.append(f"市场波动率过高: {volatility:.2%}") 

397 

398 stress_level = market_data.get('market_stress_level', 0) 

399 if stress_level > 0.5: 

400 reasons.append(f"市场压力水平较高: {stress_level:.2%}") 

401 

402 if not reasons: 

403 reasons.append("基于当前市场条件的最优调整") 

404 

405 return " | ".join(reasons)

407 def _calculate_expected_impact(self, original_limits: Dict[str, float], 

408 

409 

410 adjusted_limits: Dict[str, float], 

411 market_data: Dict[str, Any]) -> Dict[str, float]: 

412 """计算预期影响""" 

413 impact = {} 

414 

415 for limit_name in original_limits: 

416 if limit_name in adjusted_limits: 

417 original = original_limits[limit_name] 

418 adjusted = adjusted_limits[limit_name] 

419 

420 if original > 0: 

421 change_pct = (adjusted - original) / original 

422 impact[f"{limit_name}_change"] = change_pct 

423 

424 # 估算风险影响 

425 if 'position' in limit_name.lower(): 

426 impact['expected_position_impact'] = -change_pct * 0.5 # 简化的影响估算 

427 elif 'var' in limit_name.lower(): 

428 impact['expected_var_impact'] = -change_pct * 0.3 

429 elif 'loss' in limit_name.lower(): 

430 impact['expected_loss_impact'] = -change_pct * 0.8 

431 

432 # 整体风险影响 

433 avg_change = np.mean([v for k, v in impact.items() if k.endswith('_change')]) 

434 impact['overall_risk_impact'] = avg_change if not np.isnan(avg_change) else 0 

435 

436 return impact

473 def evaluate_and_adjust(self, asset_symbol: str, market_data: Dict[str, Any]) -> Optional[LimitAdjustment]: 

474 """评估并调整限额""" 

475 if not self.is_auto_adjustment_enabled: 

476 return None 

477 

478 if asset_symbol not in self.current_limits: 

479 logger.warning(f"未找到资产限额: {asset_symbol}") 

480 return None 

481 

482 current_limits = self.current_limits[asset_symbol] 

483 

484 # 计算调整 

485 adjustment = self.adjuster.calculate_adjustment( 

486 asset_symbol, current_limits, market_data 

487 ) 

488 

489 # 验证调整合理性 

490 if self._validate_adjustment(adjustment): 

491 # 应用调整 

492 self._apply_adjustment(adjustment) 

493 self.applied_adjustments.append(adjustment) 

494 self.stats['applied_adjustments'] += 1 

495 

496 logger.info(f"限额调整已应用: {asset_symbol} - {adjustment.reason}") 

497 return adjustment 

498 else: 

499 self.stats['rejected_adjustments'] += 1 

500 adjustment.status = "rejected" 

501 logger.warning(f"限额调整被拒绝: {asset_symbol} - 验证失败") 

502 

503 return None

505 def _validate_adjustment(self, adjustment: LimitAdjustment) -> bool: 

506 """验证调整合理性""" 

507 # 检查调整幅度是否过大 

508 for limit_name, factor in adjustment.adjustment_factors.items(): 

509 if factor < 0.3: # 减少超过70% 

510 logger.warning(f"调整幅度过大: {limit_name} - {factor}") 

511 return False 

512 if factor > 2.0: # 增加超过100% 

513 logger.warning(f"调整幅度过大: {limit_name} - {factor}") 

514 return False 

515 

516 # 检查调整后的限额是否合理 

517 for limit_name, new_value in adjustment.adjusted_limits.items(): 

518 if new_value <= 0: 

519 logger.warning(f"调整后限额无效: {limit_name} - {new_value}") 

520 return False 

521 

522 return True

524 def _apply_adjustment(self, adjustment: LimitAdjustment): 

525 """应用调整""" 

526 asset_symbol = adjustment.asset_symbol 

527 self.current_limits[asset_symbol] = adjustment.adjusted_limits.copy() 

528 adjustment.status = "applied" 

529 

530 self.stats['total_adjustments'] += 1 

531 

532 # 更新统计 

533 impact = adjustment.expected_impact.get('overall_risk_impact', 0) 

534 self.stats['avg_adjustment_impact'] = ( 

535 (self.stats['avg_adjustment_impact'] * (self.stats['total_adjustments'] - 1)) 

536 + impact 

537 ) / self.stats['total_adjustments']
