104 @staticmethod 

105 @jit(nopython=True) 

106 def _calculate_pressure_numba(tick_buffer: np.ndarray, window: int) -> float: 

107 """计算高频买卖压力""" 

108 if len(tick_buffer) < window: 

109 return 0.0 

110 

111 recent = tick_buffer[-window:] 

112 buy_vol = np.sum(recent[recent[:, 3] > 0, 1]) 

113 sell_vol = np.sum(recent[recent[:, 3] < 0, 1]) 

114 return (buy_vol - sell_vol) / (buy_vol + sell_vol + 1e-6) 

115 

116 def calculate_high_freq_pressure(self) -> float: 

117 """计算高频买卖压力""" 

118 return self._calculate_pressure_numba( 

119 self.tick_buffer[:self.buffer_index], 

120 self.config.window_size

123 def calculate_instant_liquidity(self, order_book: Dict[str, np.ndarray]) -> float: 

124 """计算瞬时流动性""" 

125 bid, ask = order_book["bid"], order_book["ask"] 

126 bid_vol = np.sum(bid[:self.config.depth, 1]) 

127 ask_vol = np.sum(ask[:self.config.depth, 1]) 

128 return (bid_vol + ask_vol) / 2

130 def track_large_orders(self, trades: np.ndarray) -> Dict[str, float]: 

131 """追踪大单交易""" 

132 large_buys = trades[(trades[:, 1] > 0) & (trades[:, 2] > 100000)] 

133 large_sells = trades[(trades[:, 1] < 0) & (trades[:, 2] > 100000)] 

134 

135 return { 

136 "large_buy_vol": np.sum(large_buys[:, 2]) if len(large_buys) > 0 else 0.0, 

137 "large_sell_vol": np.sum(large_sells[:, 2]) if len(large_sells) > 0 else 0.0 

138 }

140 def calculate_a_share_features(self, order_book: Dict[str, np.ndarray]) -> Dict[str, float]: 

141 """计算A股特有Level2特征""" 

142 if not self.config.a_share_specific: 

143 return {} 

144 

145 # 涨跌停板附近挂单分析 

146 upper_limit = order_book.get("upper_limit", 0) 

147 lower_limit = order_book.get("lower_limit", 0) 

148 

149 bid_pressure = np.sum(order_book["bid"][:, 0] >= upper_limit * 0.99) 

150 ask_pressure = np.sum(order_book["ask"][:, 0] <= lower_limit * 1.01) 

151 

152 return { 

153 "limit_order_pressure": (bid_pressure - ask_pressure) / self.config.depth

179 def calculate_metrics(self, order_book: Dict[str, np.ndarray]) -> Dict[str, float]: 

180 """计算基础指标""" 

181 if "bid" not in order_book or "ask" not in order_book: 

182 return {} 

183 

184 bid = order_book["bid"] 

185 ask = order_book["ask"] 

186 

187 if len(bid) == 0 or len(ask) == 0: 

188 return {} 

189 

190 # 计算买卖价差 

191 spread = ask[0, 0] - bid[0, 0] 

192 

193 # 计算买卖量 

194 bid_volume = np.sum(bid[:, 1]) 

195 ask_volume = np.sum(ask[:, 1]) 

196 

197 # 计算深度不平衡 

198 depth_imbalance = (bid_volume - ask_volume) / (bid_volume + ask_volume + 1e-6) 

199 

200 return { 

201 "spread": spread, 

202 "bid_volume": bid_volume, 

203 "ask_volume": ask_volume, 

204 "depth_imbalance": depth_imbalance 

205 }
