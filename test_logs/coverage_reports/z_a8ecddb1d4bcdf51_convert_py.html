27def _safe_logger_log(level: int, message: str) -> None: 

28 """在单测环境中安全输出日志，兼容被 mock 的 handler.level。""" 

29 seen_handlers = set() 

30 visited_loggers = set() 

31 current_logger = logger 

32 depth = 0 

33 

34 while current_logger and id(current_logger) not in visited_loggers and depth < 10: 

35 visited_loggers.add(id(current_logger)) 

36 depth += 1 

37 

38 handlers_attr = getattr(current_logger, "handlers", None) 

39 if isinstance(handlers_attr, (list, tuple, set)): 

40 handlers = list(handlers_attr) 

41 elif isinstance(handlers_attr, logging.Handler): 

42 handlers = [handlers_attr] 

43 else: 

44 handlers = [] 

45 

46 for handler in handlers: 

47 if id(handler) in seen_handlers: 

48 continue 

49 seen_handlers.add(id(handler)) 

50 level_value = getattr(handler, "level", logging.NOTSET) 

51 if not isinstance(level_value, int): 

52 try: 

53 handler.setLevel(logging.NOTSET) # type: ignore[attr-defined] 

54 except Exception: 

55 try: 

56 handler.level = logging.NOTSET # type: ignore[attr-defined] 

57 except Exception: 

58 try: 

59 handler.__dict__["level"] = logging.NOTSET # type: ignore[attr-defined] 

60 except Exception: 

61 pass 

62 if not isinstance(getattr(handler, "level", None), int): 

63 try: 

64 object.__setattr__(handler, "level", logging.NOTSET) # type: ignore[attr-defined] 

65 except Exception: 

66 pass 

67 

68 if not getattr(current_logger, "propagate", True): 

69 break 

70 parent_logger = getattr(current_logger, "parent", None) 

71 if parent_logger is None or parent_logger is current_logger: 

72 break 

73 if not isinstance(parent_logger, logging.Logger): 

74 break 

75 current_logger = parent_logger 

76 

77 try: 

78 logger.log(level, message) 

79 except TypeError: 

80 logging.getLogger(logger.name).log(level, message)

104def _apply_adjustment_factors_vectorized( 

105 data: pd.DataFrame, factor_dates: list, factor_values: list 

106) -> pd.DataFrame: 

107 """向量化应用复权因子 - 性能优化版""" 

108 price_columns = ["open", "high", "low", "close"] 

109 

110 # 成交量列（需要除以因子） 

111 volume_columns = ["volume"] 

112 

113 # 创建因子应用矩阵 

114 data_index = data.index 

115 factor_matrix = np.ones(len(data_index)) 

116 

117 # 为每个因子计算累积应用范围 

118 cumulative_factor = 1.0 

119 prev_date_idx = 0 

120 

121 for i, (date, factor) in enumerate(zip(factor_dates, factor_values)): 

122 if date in data_index: 

123 # 找到日期在数据中的位置 

124 date_idx = data_index.get_loc(date) 

125 

126 # 在前一个因子日期到当前因子日期之间的数据应用累积因子 

127 if date_idx > prev_date_idx: 

128 factor_matrix[prev_date_idx:date_idx] *= cumulative_factor 

129 

130 # 更新累积因子 

131 cumulative_factor *= factor 

132 prev_date_idx = date_idx 

133 

134 # 最后一个因子应用到剩余的所有数据 

135 if prev_date_idx < len(factor_matrix): 

136 factor_matrix[prev_date_idx:] *= cumulative_factor 

137 

138 # 向量化应用到价格列 

139 for col in price_columns: 

140 if col in data.columns: 

141 data[col] = data[col].values * factor_matrix 

142 

143 # 向量化应用到成交量列（除法） 

144 for col in volume_columns: 

145 if col in data.columns: 

146 data[col] = data[col].values / factor_matrix 

147 

148 return data

178 def calculate_limit_prices( 

179 prev_close: float, is_st: bool = False 

180 ) -> Dict[str, float]: 

181 """ 

182 计算A股涨跌停价格 

183 Args: 

184 prev_close: 前收盘价 

185 is_st: 是否为ST股票 

186 Returns: 

187 {"upper_limit": 涨停价, "lower_limit": 跌停价} 

188 """ 

189 if not isinstance(prev_close, (float, int, Decimal)): 

190 raise ValueError("prev_close必须是数值类型")

197 lower = float( 

198 Decimal(str(prev_close)) 

199 * Decimal(str(DataConvertConstants.PRICE_CALCULATION_BASE - multiplier)) 

200 ) 

201 

202 # A股价格最小变动单位为0.01元 

203 return { 

204 "upper_limit": round(upper, DataConvertConstants.PRICE_ROUNDING_DECIMALS), 

205 "lower_limit": round(lower, DataConvertConstants.PRICE_ROUNDING_DECIMALS),

311 @staticmethod 

312 def convert_frequency( 

313 data: pd.DataFrame, freq: str, agg_rules: Dict[str, str] = None 

314 ): 

315 """ 

316 转换数据频率 

317 Args: 

318 data: 原始数据(必须包含datetime索引) 

319 freq: 目标频率('1min','5min','1H','1D','1W','1M') 

320 agg_rules: 各列的聚合规则 

321 Returns: 

322 转换频率后的DataFrame 

323 """ 

324 if not isinstance(data.index, pd.DatetimeIndex): 

325 raise ValueError("数据必须包含datetime索引") 

326 

327 default_rules = { 

328 "open": "first", 

329 "high": "max", 

330 "low": "min", 

331 "close": "last", 

332 "volume": "sum", 

333 "amount": "sum", 

334 } 

335 

336 agg_rules = agg_rules or default_rules 

337 

338 # 保留原始数据中存在的列 

339 valid_cols = [col for col in agg_rules if col in data.columns] 

340 agg_rules = {col: agg_rules[col] for col in valid_cols} 

341 

342 if not valid_cols: 

343 raise ValueError("没有有效的列可以进行频率转换") 

344 

345 # 使用pandas的resample进行频率转换 

346 try: 

347 result = data.resample(freq).agg(agg_rules) 

348 return result 

349 except Exception as e: 

350 _safe_logger_log(logging.ERROR, f"频率转换失败: {e}") 

351 raise ValueError(f"不支持的频率或数据格式: {freq}")
