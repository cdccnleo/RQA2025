15 - 方法命名规范

21 @staticmethod 

22 def check_interface_implementation(interface_cls: type, implementation_cls: type) -> Dict[str, Any]: 

23 """ 

24 检查接口实现的一致性 

25 

26 Args: 

27 interface_cls: 接口类 

28 implementation_cls: 实现类 

29 

30 Returns: 

31 Dict[str, Any]: 检查结果报告 

32 """ 

33 result = { 

34 'interface': interface_cls.__name__, 

35 'implementation': implementation_cls.__name__, 

36 'is_consistent': True, 

37 'issues': [], 

38 'method_coverage': 0.0 

39 } 

40 

41 # 检查是否继承自接口 

42 if not issubclass(implementation_cls, interface_cls): 

43 result['is_consistent'] = False 

44 result['issues'].append(f'{implementation_cls.__name__} 未继承 {interface_cls.__name__}') 

45 return result 

46 

47 # 获取接口的抽象方法 

48 interface_methods = InterfaceConsistencyChecker._get_abstract_methods(interface_cls) 

49 

50 # 检查方法实现 - 需要检查实现类是否实际实现了这些方法 

51 implemented_methods = set() 

52 for method_name in interface_methods: 

53 # 检查方法是否在实现类中定义（不是继承的） 

54 if method_name in implementation_cls.__dict__ or \ 

55 (hasattr(implementation_cls, method_name) and 

56 getattr(implementation_cls, method_name) != getattr(interface_cls, method_name, None)): 

57 implemented_methods.add(method_name) 

58 else: 

59 result['is_consistent'] = False 

60 result['issues'].append(f'缺少方法实现: {method_name}') 

61 

62 # 计算覆盖率 

63 result['method_coverage'] = len(implemented_methods) / \ 

64 len(interface_methods) if interface_methods else 1.0 

65 

66 return result

68 @staticmethod 

69 def _get_abstract_methods(cls: type) -> List[str]: 

70 """获取类的所有抽象方法和属性""" 

71 abstract_members = [] 

72 for name in dir(cls): 

73 if name.startswith('_'): 

74 continue 

75 attr = getattr(cls, name) 

76 if hasattr(attr, '__isabstractmethod__') and attr.__isabstractmethod__: 

77 abstract_members.append(name) 

78 return abstract_members

99 # 对于实现类，检查方法命名 

100 methods = [name for name in dir(cls) if not name.startswith(

102 naming_issues = []

104 for method in methods:

107 method in ['component_name', 'component_type']): 

108 naming_issues.append(f'方法名不符合规范: {method}') 

109
