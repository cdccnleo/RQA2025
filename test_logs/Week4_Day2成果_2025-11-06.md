```python
    """测试正常流程"""
    @handle_monitoring_exception(operation="test")
    def test_func():
        return "success"
    
    result = test_func()
    assert result == "success"  # ✅ 正常流程不受影响

def test_handle_monitoring_exception_catches_error():
    """测试异常捕获"""
    @handle_monitoring_exception(operation="test")
    def test_func():
        raise ValueError("Test error")
    
    with pytest.raises(Exception):  # ✅ 异常被捕获
        test_func()
```

**价值**: 验证装饰器在正常/异常两种情况下都能正确工作
