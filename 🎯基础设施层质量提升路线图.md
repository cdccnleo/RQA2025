# scripts/batch_fix_mocks.py
import re
import os
from pathlib import Path

def fix_incomplete_mocks(test_file_path):
    """批量修复不完整的Mock配置"""
    with open(test_file_path, 'r', encoding='utf-8') as f:
        content = f.read()
    
    # 替换不完整的Mock为标准Mock
    patterns = [
        (r'mock = Mock\(\)', 'mock = StandardMockBuilder.create_cache_mock()'),
        (r'mock = MagicMock\(\)', 'mock = StandardMockBuilder.create_cache_mock()'),
        # ... 更多模式
    ]
    
    modified = content
    for pattern, replacement in patterns:
        modified = re.sub(pattern, replacement, modified)
    
    if modified != content:
        with open(test_file_path, 'w', encoding='utf-8') as f:
            f.write(modified)
        return True
    return False
```

# 批量修复脚本
import re

def fix_async_tests(file_path):
    """为异步测试添加装饰器"""
    with open(file_path, 'r', encoding='utf-8') as f:
        lines = f.readlines()
    
    modified_lines = []
    for i, line in enumerate(lines):
        # 如果是async def test_开头且没有@pytest.mark.asyncio
        if line.strip().startswith('async def test_'):
            prev_line = lines[i-1].strip() if i > 0 else ""
            if not prev_line.startswith('@pytest.mark.asyncio'):
                modified_lines.append('    @pytest.mark.asyncio\n')
        modified_lines.append(line)
    
    return ''.join(modified_lines)
```
